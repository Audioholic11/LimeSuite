346 void pnlQSpark::RegisterParameterChangeHandler(wxCommandEvent& event)
347 {
348  if (controlsPtr2Registers.find(event.GetEventObject()) == controlsPtr2Registers.end())
349  return; //control not found in the table
350 
351  Register reg = controlsPtr2Registers[event.GetEventObject()];
352  int mac = (reg.address!=0x17) && (rbChannelB->GetValue()) ? 0x2 : 0x1;
353  if (LMS_WriteFPGAReg(lmsControl, 0xFFFF, mac)!=0)
354  {
355  wxMessageBox(_("Write FPGA register failed"), _("Error"), wxICON_ERROR | wxOK);
356  return;
357  }
358 
359  unsigned short mask = (~(~0 << (reg.msb - reg.lsb + 1))) << reg.lsb; // creates bit mask
360 
361  uint16_t regValue;
362  LMS_ReadFPGAReg(lmsControl,reg.address,&regValue);
363 
364  regValue &= ~mask;
365  int evtVal = event.GetInt();
366  if (event.GetEventObject() == chkLB_1A || event.GetEventObject() == chkLB_1B || event.GetEventObject() == chkLB_2A || event.GetEventObject() == chkLB_2B)
367  evtVal++;
368  regValue |= ( evtVal << reg.lsb) & mask;
369 
370  if(LMS_WriteFPGAReg(lmsControl, reg.address, regValue) != 0)
371  wxMessageBox(_("Write FPGA register failed"), _("Error"), wxICON_ERROR | wxOK);
372 }

379 void pnlQSpark::OnbtnUpdateAll(wxCommandEvent& event)
380 {
381  map<wxObject*, Register>::iterator iter;
382  wxClassInfo* spinctr = wxClassInfo::FindClass("wxSpinCtrl");
383  wxClassInfo* checkboxctr = wxClassInfo::FindClass("wxCheckBox");
384  wxClassInfo* choicectr = wxClassInfo::FindClass("wxChoice");
385  if (LMS_WriteFPGAReg(lmsControl, 0xFFFF, rbChannelB->GetValue() ? 0x2 : 0x1)!= 0)
386  {
387  wxMessageBox(_("Write FPGA register failed"), _("Error"), wxICON_ERROR | wxOK);
388  return;
389  }
390  for (iter = controlsPtr2Registers.begin(); iter != controlsPtr2Registers.end(); ++iter)
391  {
392  Register reg = iter->second;
393  unsigned short mask = (~(~0 << (reg.msb - reg.lsb + 1))) << reg.lsb; // creates bit mask
394  uint16_t value;
395 
396  LMS_ReadFPGAReg(lmsControl,reg.address,&value);
397 
398  value = value & mask;
399  value = value >> reg.lsb;
400 
401  if (iter->first == chkLB_1A || iter->first == chkLB_1B || iter->first == chkLB_2A || iter->first == chkLB_2B)
402  value = value == 2 ? 1 : 0;
403  if (iter->first->IsKindOf(spinctr))
404  reinterpret_cast<wxSpinCtrl*>(iter->first)->SetValue(value);
405  else if(iter->first->IsKindOf(checkboxctr))
406  reinterpret_cast<wxCheckBox*>(iter->first)->SetValue(value);
407  else if(iter->first->IsKindOf(choicectr))
408  reinterpret_cast<wxComboBox*>(iter->first)->SetSelection(value);
409  }
410 
411  double refClk_MHz, ncoFreq_MHz;
412  txtPllFreqTxMHz->GetValue().ToDouble(&refClk_MHz);
413  uint32_t fcw = 0;
414  vector<uint32_t> addrs = { 0x008E, 0x008F };
415 
416  for (size_t i = 0; i <addrs.size(); i++)
417  {
418  uint16_t value;
419  if (LMS_ReadFPGAReg(lmsControl, addrs[i], &value) != 0)
420  {
421  wxMessageBox(_("Read FPGA register failed"), _("Error"), wxICON_ERROR | wxOK);
422  return;
423  }
424  fcw <<= 16;
425  fcw += value;
426  }
427  ncoFreq_MHz = fcw * refClk_MHz / 4294967296;
428  txtNcoFreq->SetValue(wxString::Format("%1.3f", ncoFreq_MHz));
429 }
