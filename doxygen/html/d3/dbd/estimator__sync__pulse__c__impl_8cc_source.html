75  int
76  estimator_sync_pulse_c_impl::work (int noutput_items,
77  gr_vector_int &ninput_items,
78  gr_vector_const_void_star &input_items,
79  gr_vector_void_star &output_items)
80  {
81  const gr_complex *in_tx = (const gr_complex *) input_items[0];
82  const gr_complex *in_rx = (const gr_complex *) input_items[1];
83 
84  // Resize buffers
85  int ninput_items_min = std::min(ninput_items[0],ninput_items[1]);
86  if(d_noutput_items_vec!=ninput_items_min){
87  d_in_tx_real.resize(ninput_items_min);
88  d_in_rx_real.resize(ninput_items_min);
89  }
90 
91  // Copy complex to float and get abs
92  for(int k=0; k<ninput_items_min; k++){
93  d_in_tx_real[k] = std::abs(in_tx[k]);
94  d_in_rx_real[k] = std::abs(in_rx[k]);
95  }
96 
97  // Do cross correlation and find max
98  float xcorr, xcorr_hold;
99  int num_delay;
100  num_delay = -1;
101  xcorr_hold = -1;
102 
103  for(int k=0; k<d_num_xcorr; k++){
104  volk_32f_x2_dot_prod_32f(&xcorr,&d_in_tx_real[0],&d_in_rx_real[0]+k,ninput_items_min-k);
105  if(xcorr>xcorr_hold){
106  xcorr_hold = xcorr;
107  num_delay = k;
108  }
109  }
110 
111  // Get rx_time tag
112  std::vector< tag_t > tags;
113  pmt::pmt_t ptimestamp, msg_out, pxcorr;
114  get_tags_in_range(tags,1,nitems_read(1),nitems_read(1)+1,pmt::string_to_symbol("rx_time"));
115 
116  // Setup msg pmt
117  if(tags.size()>0) ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),tags[0].value);
118  else ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),pmt::make_tuple(pmt::from_uint64(0),pmt::from_double(-1))); // if no timetag is found, set to 0 and frac_sec to -1
119 
120  pxcorr = pmt::list2(pmt::string_to_symbol("sync_pulse"),pmt::from_long(std::abs(num_delay)));
121  msg_out = pmt::list2(ptimestamp,pxcorr);
122 
123  // Publish message
124  message_port_pub(d_port_id,msg_out);
125 
126  // Tell runtime system how many output items we produced.
127  return 0;
128  }
