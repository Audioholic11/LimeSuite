412  xB += xBn * mult;
413  }

590  }
591 
592  //configure FPGA on first start, or disable FPGA when not streaming
593  if((needTx || needRx) && (!txThread.joinable()) && (!rxThread.joinable()))
594  {
595  dataPort->WriteRegister(0xFFFF, 1 << chipId);
596  if (mRxStreams[0].used && mRxStreams[1].used)
597  AlignRxRF(true);
598  //enable FPGA streaming
599  fpga->StopStreaming();
600  fpga->ResetTimestamp();
601  rxLastTimestamp.store(0);
602  //Clear device stream buffers
603  dataPort->ResetStreamBuffers();
604 
605  //enable MIMO mode, 12 bit compressed values
606  dataLinkFormat = StreamConfig::FMT_INT12;
607  //by default use 12 bit compressed, adjust link format for stream
608 
609  for(auto &i : mRxStreams)
610  if(i.used && i.config.format != StreamConfig::FMT_INT12)
611  {
612  dataLinkFormat = StreamConfig::FMT_INT16;
613  break;
614  }
615 
616  for(auto &i : mTxStreams)
617  if(i.used && i.config.format != StreamConfig::FMT_INT12)
618  {
619  dataLinkFormat = StreamConfig::FMT_INT16;
620  break;
621  }
622 
623  for(auto &i : mRxStreams)
624  if (i.used)
625  i.config.linkFormat = dataLinkFormat;
626  for(auto &i : mTxStreams)
627  if (i.used)
628  i.config.linkFormat = dataLinkFormat;
629 
630  const uint16_t smpl_width = dataLinkFormat == StreamConfig::FMT_INT12 ? 2 : 0;
631  uint16_t mode = 0x0100;
632 
633  if (lms->Get_SPI_Reg_bits(LMS7param(LML1_SISODDR)))
634  mode = 0x0040;
635  else if (lms->Get_SPI_Reg_bits(LMS7param(LML1_TRXIQPULSE)))
636  mode = 0x0180;
637 
638  dataPort->WriteRegister(0x0008, mode | smpl_width);
639 
640  const uint16_t channelEnables = (mRxStreams[0].used||mTxStreams[0].used) + 2 * (mRxStreams[1].used||mTxStreams[1].used);
641  dataPort->WriteRegister(0x0007, channelEnables);
642 
643  uint32_t reg9;
644  dataPort->ReadRegister(0x0009, reg9);
645  const uint32_t addr[] = {0x0009, 0x0009};
646  const uint32_t data[] = {reg9 | (5 << 1), reg9 & ~(5 << 1)};
647  fpga->StartStreaming();
648  dataPort->WriteRegisters(addr, data, 2);
649  }
650  else if(not needTx and not needRx)
651  {
652  //disable FPGA streaming
653  dataPort->WriteRegister(0xFFFF, 1 << chipId);
654  fpga->StopStreaming();
655  }
656 
657  //FPGA should be configured and activated, start needed threads
658  if(needRx && (!rxThread.joinable()))
659  {
660  terminateRx.store(false);
661  auto RxLoopFunction = std::bind(&Streamer::ReceivePacketsLoop, this);
662  rxThread = std::thread(RxLoopFunction);
663  }
664  if(needTx && (!txThread.joinable()))
665  {
666  dataPort->WriteRegister(0xFFFF, 1 << chipId);
667  dataPort->WriteRegister(0xD, 0); //stop WFM
668  terminateTx.store(false);
669  auto TxLoopFunction = std::bind(&Streamer::TransmitPacketsLoop, this);
670  txThread = std::thread(TxLoopFunction);
671  }
672  return 0;

675 void Streamer::TransmitPacketsLoop()
676 {
677  //at this point FPGA has to be already configured to output samples
678  const uint8_t maxChannelCount = 2;
679  const uint8_t chCount = streamSize;
680  const bool packed = dataLinkFormat == StreamConfig::FMT_INT12;
681  const int epIndex = chipId;
682  const uint8_t buffersCount = dataPort->GetBuffersCount();
683  const uint8_t packetsToBatch = dataPort->CheckStreamSize(rxBatchSize);
684  const uint32_t bufferSize = packetsToBatch*sizeof(FPGA_DataPacket);
685  const uint32_t popTimeout_ms = 500;
686 
687  const int maxSamplesBatch = (packed ? samples12InPkt:samples16InPkt)/chCount;
688  std::vector<int> handles(buffersCount, 0);
689  std::vector<bool> bufferUsed(buffersCount, 0);
690  std::vector<uint32_t> bytesToSend(buffersCount, 0);
691  std::vector<complex16_t> samples[maxChannelCount];
692  std::vector<char> buffers;
693  try
694  {
695  for(int i=0; i<chCount; ++i)
696  samples[i].resize(maxSamplesBatch);
697  buffers.resize(buffersCount*bufferSize, 0);
698  }
699  catch (const std::bad_alloc& ex) //not enough memory for buffers
700  {
701  return lime::error("Error allocating Tx buffers, not enough memory");
702  }
703 
704  long totalBytesSent = 0;
705  auto t1 = std::chrono::high_resolution_clock::now();
706  auto t2 = t1;
707  bool end_burst = false;
708  uint8_t bi = 0; //buffer index
709  while (terminateTx.load() != true)
710  {
711  if (bufferUsed[bi])
712  {
713  if (dataPort->WaitForSending(handles[bi], 1000) == true)
714  {
715  unsigned bytesSent = dataPort->FinishDataSending(&buffers[bi*bufferSize], bytesToSend[bi], handles[bi]);
716 
717  if (bytesSent != bytesToSend[bi])
718  {
719  for (auto &value : mTxStreams)
720  if (value.used && value.mActive)
721  value.overflow++;
722  }
723  else
724  totalBytesSent += bytesSent;
725  bufferUsed[bi] = false;
726  }
727  else
728  {
729  txDataRate_Bps.store(totalBytesSent);
730  totalBytesSent = 0;
731  continue;
732  }
733  }
734 
735  FPGA_DataPacket* pkt = reinterpret_cast<FPGA_DataPacket*>(&buffers[bi*bufferSize]);
736  int i=0;
737  do
738  {
739  bool has_samples = false;
740  StreamChannel::Metadata meta = {0, 0};
741  for(int ch=0; ch<maxChannelCount; ++ch)
742  {
743  if (!mTxStreams[ch].used)
744  continue;
745  const int ind = chCount == maxChannelCount ? ch : 0;
746  if (mTxStreams[ch].mActive==false)
747  {
748  memset(&samples[ind][0],0,maxSamplesBatch*sizeof(complex16_t));
749  continue;
750  }
751  int samplesPopped = mTxStreams[ch].Read(samples[ind].data(), maxSamplesBatch, &meta, popTimeout_ms);
752  if (samplesPopped != maxSamplesBatch)
753  {
754  if ((!end_burst) && !(meta.flags & RingFIFO::END_BURST))
755  {
756  mTxStreams[ch].underflow++;
757  lime::warning("popping from TX, samples popped %i/%i", samplesPopped, maxSamplesBatch);
758  continue;
759  }
760  memset(&samples[ind][samplesPopped],0,(maxSamplesBatch-samplesPopped)*sizeof(complex16_t));
761  }
762  has_samples = true;
763  }
764 
765  if (!has_samples)
766  break;
767 
768  end_burst = (meta.flags & RingFIFO::END_BURST);
769  pkt[i].counter = meta.timestamp;
770  pkt[i].reserved[0] = 0;
771  //by default ignore timestamps
772  const int ignoreTimestamp = !(meta.flags & RingFIFO::SYNC_TIMESTAMP);
773  pkt[i].reserved[0] |= ((int)ignoreTimestamp << 4); //ignore timestamp
774 
775  std::vector<complex16_t*> src(chCount);
776  for(uint8_t c=0; c<chCount; ++c)
777  src[c] = (samples[c].data());
778  uint8_t* const dataStart = (uint8_t*)pkt[i].data;
779  FPGA::Samples2FPGAPacketPayload(src.data(), maxSamplesBatch, chCount==2, packed, dataStart);
780 
781  }while(++i<packetsToBatch && end_burst == false);
782 
783  if(terminateTx.load() == true) //early termination
784  break;
785 
786  if (i)
787  {
788  bytesToSend[bi] = i*sizeof(FPGA_DataPacket);
789  handles[bi] = dataPort->BeginDataSending(&buffers[bi*bufferSize], bytesToSend[bi], epIndex);
790  txLastTimestamp.store(pkt[i-1].counter+maxSamplesBatch-1); //timestamp of the last sample that was sent to HW
791  bufferUsed[bi] = true;
792  bi = (bi + 1) & (buffersCount-1);
793  }
794 
795  t2 = std::chrono::high_resolution_clock::now();
796  auto timePeriod = std::chrono::duration_cast<std::chrono::milliseconds>(t2 - t1).count();
797  if (timePeriod >= 1000)
798  {
799  //total number of bytes sent per second
800  float dataRate = 1000.0*totalBytesSent / timePeriod;
801  txDataRate_Bps.store(dataRate);
802  totalBytesSent = 0;
803  t1 = t2;
804 #ifndef NDEBUG
805  printf("Tx: %.3f MB/s\n", dataRate / 1000000.0);
806 #endif
807  }
808  }
809 
810  // Wait for all the queued requests to be cancelled
811  dataPort->AbortSending(epIndex);
812  txDataRate_Bps.store(0);
813 }

818 void Streamer::ReceivePacketsLoop()
819 {
820  //at this point FPGA has to be already configured to output samples
821  const uint8_t maxChannelCount = 2;
822  const uint8_t chCount = streamSize;
823  const bool packed = dataLinkFormat == StreamConfig::FMT_INT12;
824  const uint32_t samplesInPacket = (packed ? samples12InPkt : samples16InPkt)/chCount;
825 
826  const int epIndex = chipId;
827  const uint8_t buffersCount = dataPort->GetBuffersCount();
828  const uint8_t packetsToBatch = dataPort->CheckStreamSize(rxBatchSize);
829  const uint32_t bufferSize = packetsToBatch*sizeof(FPGA_DataPacket);
830  std::vector<int> handles(buffersCount, 0);
831  std::vector<char>buffers(buffersCount*bufferSize, 0);
832  std::vector<StreamChannel::Frame> chFrames;
833  try
834  {
835  chFrames.resize(chCount);
836  }
837  catch (const std::bad_alloc &ex)
838  {
839  lime::error("Error allocating Rx buffers, not enough memory");
840  return;
841  }
842 
843  for (int i = 0; i<buffersCount; ++i)
844  handles[i] = dataPort->BeginDataReading(&buffers[i*bufferSize], bufferSize, epIndex);
845 
846  int bi = 0;
847  unsigned long totalBytesReceived = 0; //for data rate calculation
848 
849  auto t1 = std::chrono::high_resolution_clock::now();
850  auto t2 = t1;
851 
852  std::mutex txFlagsLock;
853  std::condition_variable resetTxFlags;
854  //worker thread for reseting late Tx packet flags
855  std::thread txReset([](IConnection* port,
856  std::atomic<bool> *terminate,
857  std::mutex *spiLock,
858  std::condition_variable *doWork)
859  {
860  uint32_t reg9;
861  port->ReadRegister(0x0009, reg9);
862  const uint32_t addr[] = {0x0009, 0x0009};
863  const uint32_t data[] = {reg9 | (5 << 1), reg9 & ~(5 << 1)};
864  while (not terminate->load())
865  {
866  std::unique_lock<std::mutex> lck(*spiLock);
867  doWork->wait(lck);
868  port->WriteRegisters(addr, data, 2);
869  }
870  }, dataPort, &terminateRx, &txFlagsLock, &resetTxFlags);
871 
872  int resetFlagsDelay = 0;
873  uint64_t prevTs = 0;
874  while (terminateRx.load() == false)
875  {
876  int32_t bytesReceived = 0;
877  if(handles[bi] >= 0)
878  {
879  if (dataPort->WaitForReading(handles[bi], 1000) == true)
880  {
881  bytesReceived = dataPort->FinishDataReading(&buffers[bi*bufferSize], bufferSize, handles[bi]);
882  totalBytesReceived += bytesReceived;
883  if (bytesReceived != int32_t(bufferSize)) //data should come in full sized packets
884  for(auto &value: mRxStreams)
885  if (value.used && value.mActive)
886  value.underflow++;
887  }
888  else
889  {
890  rxDataRate_Bps.store(totalBytesReceived);
891  totalBytesReceived = 0;
892  continue;
893  }
894  }
895  bool txLate=false;
896  for (uint8_t pktIndex = 0; pktIndex < bytesReceived / sizeof(FPGA_DataPacket); ++pktIndex)
897  {
898  const FPGA_DataPacket* pkt = (FPGA_DataPacket*)&buffers[bi*bufferSize];
899  const uint8_t byte0 = pkt[pktIndex].reserved[0];
900 
901  if ((byte0 & (1 << 3)) != 0 && !txLate) //report only once per batch
902  {
903  txLate = true;
904  if(resetFlagsDelay > 0)
905  --resetFlagsDelay;
906  else
907  {
908  lime::warning("L");
909  resetTxFlags.notify_one();
910  resetFlagsDelay = buffersCount;
911  for(auto &value: mTxStreams)
912  if (value.used && value.mActive)
913  value.pktLost++;
914  }
915  }
916  uint8_t* pktStart = (uint8_t*)pkt[pktIndex].data;
917  if(pkt[pktIndex].counter - prevTs != samplesInPacket && pkt[pktIndex].counter != prevTs)
918  {
919  int packetLoss = ((pkt[pktIndex].counter - prevTs)/samplesInPacket)-1;
920  for(auto &value: mRxStreams)
921  if (value.used && value.mActive)
922  value.pktLost += packetLoss;
923  }
924  prevTs = pkt[pktIndex].counter;
925  rxLastTimestamp.store(prevTs);
926 
927  chirpLastTimeStamp.store(pkt[pktIndex].ftr0);
928  chirpLastTimePeriod.store(pkt[pktIndex].ftr1);
929 
930  //parse samples
931  std::vector<complex16_t*> dest(chCount);
932  for(uint8_t c=0; c<chCount; ++c)
933  dest[c] = (chFrames[c].samples);
934  int samplesCount = FPGA::FPGAPacketPayload2Samples(pktStart, dataLength, chCount==2, packed, dest.data());
935 
936  for(int ch=0; ch<maxChannelCount; ++ch)
937  {
938  if (mRxStreams[ch].used==false || mRxStreams[ch].mActive==false)
939  continue;
940  const int ind = chCount == maxChannelCount ? ch : 0;
941  StreamChannel::Metadata meta;
942  meta.timestamp = pkt[pktIndex].counter;
943  meta.lastchirp_timestamp = pkt[pktIndex].ftr0;
944  meta.chirptime = pkt[pktIndex].ftr1;
945  meta.flags = RingFIFO::OVERWRITE_OLD | RingFIFO::SYNC_TIMESTAMP;
946  int samplesPushed = mRxStreams[ch].Write((const void*)chFrames[ind].samples, samplesCount, &meta, 100);
947  if(samplesPushed != samplesCount)
948  mRxStreams[ch].overflow++;
949  }
950  }
951  // Re-submit this request to keep the queue full
952  handles[bi] = dataPort->BeginDataReading(&buffers[bi*bufferSize], bufferSize, epIndex);
953  bi = (bi + 1) & (buffersCount-1);
954 
955  t2 = std::chrono::high_resolution_clock::now();
956  auto timePeriod = std::chrono::duration_cast<std::chrono::milliseconds>(t2 - t1).count();
957  if (timePeriod >= 1000)
958  {
959  t1 = t2;
960  //total number of bytes sent per second
961  double dataRate = 1000.0*totalBytesReceived / timePeriod;
962 #ifndef NDEBUG
963  printf("Rx: %.3f MB/s\n", dataRate / 1000000.0);
964 #endif
965  totalBytesReceived = 0;
966  rxDataRate_Bps.store((uint32_t)dataRate);
967  }
968  }
969  dataPort->AbortReading(epIndex);
970  resetTxFlags.notify_one();
971  txReset.join();
972  rxDataRate_Bps.store(0);
973 }
974
