70  void
71  qtgui_scatter_plot_impl::handle_msg(pmt::pmt_t msg)
72  {
73  size_t size_msg;
74  d_x.clear();
75  d_y.clear();
76 
77  size_msg = pmt::length(msg);
78  pmt::pmt_t msg_part;
79  bool item_found_x, item_found_y;
80  // Go through msg and search for key symbols "range" and "velocity" (or other keys!) and get data
81  item_found_x = false;
82  item_found_y = false;
83  for(int k=0; k<size_msg; k++){
84  msg_part = pmt::nth(k,msg);
85  if(pmt::symbol_to_string(pmt::nth(0,msg_part))==d_label_x.c_str()){
86  d_x = pmt::f32vector_elements(pmt::nth(1,msg_part));
87  item_found_x = true;
88  d_xy_read = false;
89  }
90  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))==d_label_y.c_str()){
91  d_y = pmt::f32vector_elements(pmt::nth(1,msg_part));
92  item_found_y = true;
93  d_xy_read = false;
94  }
95  }
96  // Fill zeros in vector if only one label is found
97  if(not(item_found_x)&&item_found_y){
98  d_x.resize(d_y.size());
99  }
100  if(item_found_x&&not(item_found_y)){
101  d_y.resize(d_x.size());
102  }
103  if(not(item_found_x)&&not(item_found_y)){
104  throw std::runtime_error("All identifiers (symbols) not found"); // throw exception if no label found
105  }
106  }
