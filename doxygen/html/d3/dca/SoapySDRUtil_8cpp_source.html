117 /***********************************************************************
118  * Find devices and print args
119  **********************************************************************/
120 static int findDevices(const std::string &argStr, const bool sparse)
121 {
122  const auto results = SoapySDR::Device::enumerate(argStr);
123  if (sparse)
124  {
125  std::vector<std::string> sparseResults;
126  for (size_t i = 0; i < results.size(); i++)
127  {
128  const auto it = results[i].find("label");
129  if (it != results[i].end()) sparseResults.push_back(it->second);
130  else sparseResults.push_back(SoapySDR::KwargsToString(results[i]));
131  }
132  std::sort(sparseResults.begin(), sparseResults.end());
133  for (size_t i = 0; i < sparseResults.size(); i++)
134  {
135  std::cout << i << ": " << sparseResults[i] << std::endl;
136  }
137  }
138  else
139  {
140  for (size_t i = 0; i < results.size(); i++)
141  {
142  std::cout << "Found device " << i << std::endl;
143  for (const auto &it : results[i])
144  {
145  std::cout << " " << it.first << " = " << it.second << std::endl;
146  }
147  std::cout << std::endl;
148  }
149  if (results.empty()) std::cerr << "No devices found!" << std::endl;
150  else std::cout << std::endl;
151  }
152  return results.empty()?EXIT_FAILURE:EXIT_SUCCESS;
153 }
