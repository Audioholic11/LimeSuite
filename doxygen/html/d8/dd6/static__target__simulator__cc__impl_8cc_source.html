80  void
81  static_target_simulator_cc_impl::setup_targets(
82  std::vector<float> range,
83  std::vector<float> velocity,
84  std::vector<float> rcs,
85  std::vector<float> azimuth,
86  std::vector<float> position_rx,
87  int samp_rate,
88  float center_freq,
89  float self_coupling_db,
90  bool rndm_phaseshift,
91  bool self_coupling
92  ){
93  d_range = range;
94  d_velocity = velocity;
95  d_rcs = rcs;
96  d_azimuth = azimuth;
97  d_position_rx = position_rx;
98  d_center_freq = center_freq; // center frequency of simulated hardware for doppler estimation
99  d_samp_rate = samp_rate;
100  d_hold_noutput = -1;
101  d_rndm_phaseshift = rndm_phaseshift;
102  d_self_coupling = self_coupling;
103  d_self_coupling_db = self_coupling_db;
104 
105  // Setup rx_time tag
106  d_key = pmt::string_to_symbol("rx_time");
107  d_srcid = pmt::string_to_symbol("stat_targ_sim");
108 
109  // Get num targets
110  d_num_targets = range.size(); // FIXME: throw exceptions for len(range)!=len(velocity)!=...
111 
112  // Get doppler frequencies
113  d_doppler.resize(d_num_targets);
114  d_filt_doppler.resize(d_num_targets);
115  for(int k=0; k<d_num_targets; k++) d_doppler[k] = 2*d_velocity[k]*d_center_freq/c_light;
116 
117  // Get timeshifts
118  d_timeshift.resize(d_num_targets);
119  d_timeshift_azimuth.resize(d_position_rx.size());
120  d_filt_time.resize(d_num_targets);
121  d_filt_time_azimuth.resize(d_position_rx.size());
122  for(int l=0; l<d_position_rx.size(); l++){
123  d_filt_time_azimuth[l].resize(d_num_targets);
124  d_timeshift_azimuth[l].resize(d_num_targets);
125  for(int k=0; k<d_num_targets; k++){
126  d_timeshift_azimuth[l][k] = d_position_rx[l]*std::sin(d_azimuth[k]*M_PI/180.0);
127  }
128  }
129  for(int k=0; k<d_num_targets; k++) d_timeshift[k] = 2.0*d_range[k]/c_light;
130 
131  // Get signal amplitude of reflection with free space path loss and rcs (radar equation)
132  d_scale_ampl.resize(d_num_targets);
133  for (int k = 0; k < d_num_targets; k++) {
134  // Factor out all terms out of the sqrt except the RCS:
135  d_scale_ampl[k] = c_light
136  * std::sqrt(d_rcs[k])
137  / FOUR_PI_CUBED_SQRT
138  / (d_range[k] * d_range[k])
139  / d_center_freq;
140  }
141 
142  if(d_rndm_phaseshift){
143  // Resize phase shift filter
144  d_filt_phase.resize(d_num_targets);
145 
146  // Setup random numbers
147  std::srand(std::time(NULL)); // initial with time
148  }
149  }

158  int
159  static_target_simulator_cc_impl::work(
160  int noutput_items,
161  gr_vector_int &ninput_items,
162  gr_vector_const_void_star &input_items,
163  gr_vector_void_star &output_items)
164  {
165  const gr_complex *in = (const gr_complex *) input_items[0];
166  gr_complex *out;
167 
168  // Set output items to tagged stream length
169  noutput_items = ninput_items[0];
170 
171  // Check if new filter, buffer or fft plan is necessary
172  if(d_hold_noutput!=noutput_items){
173  // Set length buffer in loop
174  d_hold_in.resize(noutput_items);
175 
176  // Setup fft and ifft
177  d_in_fft.resize(noutput_items);
178  d_fft_plan = fftwf_plan_dft_1d(noutput_items, reinterpret_cast<fftwf_complex *>(&d_hold_in[0]),
179  reinterpret_cast<fftwf_complex *>(&d_in_fft[0]), FFTW_FORWARD, FFTW_ESTIMATE);
180  d_ifft_plan = fftwf_plan_dft_1d(noutput_items, reinterpret_cast<fftwf_complex *>(&d_in_fft[0]),
181  reinterpret_cast<fftwf_complex *>(&d_hold_in[0]), FFTW_BACKWARD, FFTW_ESTIMATE);
182 
183  // Setup frequency vector for shift in frequency domain
184  d_freq.resize(noutput_items);
185  for(int i=0; i<noutput_items; i++){
186  if(i<noutput_items/2) d_freq[i] = i*(float)d_samp_rate/(float)noutput_items; // zero to samp_rate/2
187  else d_freq[i] = i*(float)d_samp_rate/(float)noutput_items - (float)d_samp_rate; // -samp_rate/2 to zero
188  }
189 
190  // Setup freq and time shift filter, resize phase shift filter
191  for(int k=0; k<d_num_targets; k++){
192  d_filt_doppler[k].resize(noutput_items);
193  if(d_rndm_phaseshift) d_filt_phase[k].resize(noutput_items);
194 
195  d_phase_doppler = 0;
196  for(int i=0; i<noutput_items; i++){
197  // Doppler shift filter and rescaling amplitude with rcs
198  d_filt_doppler[k][i] = std::exp(d_phase_doppler)*d_scale_ampl[k];
199  d_phase_doppler = 1j*std::fmod(std::imag(d_phase_doppler)+2*M_PI*d_doppler[k]/(float)d_samp_rate,2*M_PI); // integrate phase (with plus!)
200  }
201 
202  d_filt_time[k].resize(noutput_items);
203  d_phase_time = 0;
204  for(int i=0; i<noutput_items; i++){
205  // Time shift filter, uses target range
206  d_phase_time = 1j*std::fmod(2*M_PI*(d_timeshift[k]) // range time shift
207  *d_freq[i],2*M_PI); // integrate phase (with minus!)
208  d_filt_time[k][i] = std::exp(-d_phase_time)/(float)noutput_items; // div with noutput_item to correct amplitude after fft->ifft
209  }
210 
211  for(int l=0; l<d_position_rx.size(); l++){ // Do time shift filter with azimuth and position, there are two time shift filters to avoid problems with significant digits of float
212  d_filt_time_azimuth[l][k].resize(noutput_items);
213  d_phase_time = 0;
214  for(int i=0; i<noutput_items; i++){
215  // Time shift filter, uses azimuth and RX position
216  d_phase_time = 1j*std::fmod(2*M_PI*(d_timeshift_azimuth[l][k]) // azimuth time shift
217  *d_freq[i],2*M_PI); // integrate phase (with minus!)
218  d_filt_time_azimuth[l][k][i] = std::exp(-d_phase_time); // do not div with noutput_items, is done with range timeshift filter
219  }
220  }
221  }
222 
223  // Resize hold of noutput_items
224  d_hold_noutput = noutput_items;
225  }
226 
227  // Setup random phase shift
228  if(d_rndm_phaseshift){
229  gr_complex phase_random_hold;
230  for(int k=0; k<d_num_targets; k++){
231  phase_random_hold = 1j*2*M_PI*float((std::rand()%1000+1)/1000.0);
232  d_phase_random = std::exp(phase_random_hold);
233  std::fill_n(&d_filt_phase[k][0],noutput_items,d_phase_random);
234  }
235  }
236 
237  // Go through RXs
238  for(int l=0; l<d_position_rx.size(); l++){
239  // Setup pointer on output buffer
240  out = (gr_complex *) output_items[l];
241 
242  // Set rx_time tag
243  d_time_sec = nitems_written(l)/d_samp_rate;
244  d_time_frac_sec = nitems_written(l)/(float)d_samp_rate-d_time_sec;
245  d_val = pmt::make_tuple
246  (pmt::from_uint64(d_time_sec),
247  pmt::from_double(d_time_frac_sec)); // FIXME: correct implementation?
248  add_item_tag(l, nitems_written(l), d_key, d_val, d_srcid);
249 
250  // Set output to zero
251  std::memset(out, 0, noutput_items*sizeof(gr_complex));
252 
253  // Go through targets and apply filters
254  for(int k=0; k<d_num_targets; k++){
255  // Add doppler shift
256  volk_32fc_x2_multiply_32fc(&d_hold_in[0], in, &d_filt_doppler[k][0], noutput_items); // add doppler shift with rescaled amplitude
257 
258  // Add time shift
259  fftwf_execute(d_fft_plan); // go to freq domain
260  volk_32fc_x2_multiply_32fc(&d_in_fft[0], &d_in_fft[0], &d_filt_time[k][0], noutput_items); // add timeshift with multiply exp-func in freq domain (range)
261  volk_32fc_x2_multiply_32fc(&d_in_fft[0], &d_in_fft[0], &d_filt_time_azimuth[l][k][0], noutput_items); // add timeshift with multiply exp-func in freq domain (rx position with azimuth and range)
262  fftwf_execute(d_ifft_plan); // back in time domain
263 
264  if(d_rndm_phaseshift){
265  // Add random phase shift
266  volk_32fc_x2_multiply_32fc(&d_hold_in[0], &d_hold_in[0], &d_filt_phase[k][0], noutput_items); // add random phase shift
267  }
268 
269  // Add to output
270  for(int i=0; i<noutput_items; i++) out[i] += d_hold_in[i];
271  }
272 
273  // Add self coupling
274  if(d_self_coupling){
275  for(int i=0; i<noutput_items; i++) out[i] += (gr_complex)pow(10,d_self_coupling_db/20.0)*in[i]; // d_self_coupling_db gives scaling of power
276  }
277  }
278 
279  // Tell runtime system how many output items we produced.
280  return noutput_items;
281  }
