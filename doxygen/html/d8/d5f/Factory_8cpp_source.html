81 SoapySDR::Device* SoapySDR::Device::make(const Kwargs &inputArgs)
82 {
83  Device *device = nullptr;
84 
85  //the arguments may have already come from enumerate and been used to open a device
86  device = getDeviceFromTable(inputArgs);
87  if (device != nullptr) return device;
88 
89  //otherwise the args must always come from an enumeration result
90  Kwargs discoveredArgs;
91  const auto results = Device::enumerate(inputArgs);
92  if (not results.empty()) discoveredArgs = results.front();
93 
94  //check the device table for an already allocated device
95  device = getDeviceFromTable(discoveredArgs);
96  if (device != nullptr) return device;
97 
98  //load the enumeration args with missing keys from the make argument
99  Kwargs hybridArgs = discoveredArgs;
100  for (const auto &it : inputArgs)
101  {
102  if (hybridArgs.count(it.first) == 0) hybridArgs[it.first] = it.second;
103  }
104 
105  //lock during device construction
106  //make itself can be parallelized, but we need to keep track of in-process factories
107  //so that other calling threads with the same args can wait on the result
108  std::lock_guard<std::recursive_mutex> lock(getFactoryMutex());
109 
110  //loop through make functions and call on module match
111  for (const auto &it : Registry::listMakeFunctions())
112  {
113  if (hybridArgs.count("driver") != 0 and hybridArgs.at("driver") != it.first) continue;
114  device = it.second(hybridArgs);
115  break;
116  }
117  if (device == nullptr) throw std::runtime_error("SoapySDR::Device::make() no match");
118 
119  //store into the table
120  getDeviceTable()[discoveredArgs] = device;
121  getDeviceCounts()[device]++;
122 
123  return device;
124 }
