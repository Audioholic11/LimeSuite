275 void ADF4002::CalculateRN()
276 {
277 
278  double x = txtFref*1000000;
279  double y = txtFvco*1000000;
280  double Fcomp;
281  while((x!=0) && (y!=0))
282  {
283  if(x >= y)
284  {
285  x = fmod(x, y);
286  }
287  else
288  {
289  y = fmod(y, x);
290  };
291  };
292 
293  Fcomp = (x + y)/1000000.0;
294  int R = (int)((txtFref/Fcomp)+0.5);
295  int N = (int)((txtFvco / Fcomp) + 0.5);
296 
297  txtRCnt = R;
298  txtNCnt = N;
299 
300  lblFcomp = Fcomp;
301  double dFvco = 0;
302  if(txtRCnt != 0)
303  dFvco = txtNCnt * (txtFref)/txtRCnt;
304  lblFvco = dFvco;
