102  tracking_singletarget_impl::handle_msg(pmt::pmt_t msg)
103  {
104  size_t size_msg;

107  velocity_found = false;

109  std::vector<pmt::pmt_t> store_msg;
110 
111  size_msg = pmt::length(msg);
112  pmt::pmt_t msg_part;
113  // Go through msg and search for key symbols "range" and "velocity" (or other keys!) and get data
114  // This is singletarget tracker! Taken only first item of vector
115  // FIXME: error handling for false input? e.g. multi target input (throw warning)
116  std::vector<float> vec_pmt;
117  d_is_empty = false; // check if range or velocity is empty
118  for(int k=0; k<size_msg; k++){ // FIXME: errorhandling for wrong input?
119  msg_part = pmt::nth(k,msg);
120  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="range"){
121  vec_pmt = (pmt::f32vector_elements(pmt::nth(1,msg_part)));
122  if(vec_pmt.size()!=0)d_range_meas = vec_pmt[0];
123  else d_is_empty = true;
124  range_found = true;
125  }
126  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="velocity"){
127  vec_pmt = (pmt::f32vector_elements(pmt::nth(1,msg_part)));
128  if(vec_pmt.size()!=0) d_velocity_meas = vec_pmt[0];
129  else d_is_empty = true;

131  }
132  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="rx_time"){
133  d_time = (float)pmt::to_uint64(pmt::tuple_ref(pmt::nth(1,msg_part),0)) + pmt::to_double(pmt::tuple_ref(pmt::nth(1,msg_part),1));
134  time_found = true;
135  store_msg.push_back(msg_part); // store msg, will not be changed
136  }
137  else{
138  // Store msg part for repacking if no symbol is found
139  store_msg.push_back(msg_part);
140  }
141  }
142 
143  // Do tracking if both variables are found, else throw error
144  bool state_tracking;
145  if(range_found&&velocity_found&&time_found) state_tracking = tracking();
146  else throw std::runtime_error("range or velocity or time not found");
147 
148  if(state_tracking){
149  // Repack msg
150  pmt::pmt_t msg_out;
151  if(store_msg.size()!=0){ // if msgs are stored add them
152  msg_out = pmt::list1(store_msg[0]);

154  msg_out = pmt::list_add(msg_out, store_msg[k]);

186  void // begin kalman-filter
187  tracking_singletarget_impl::filter_kalman(){
188  float a, b, c, d, factor;
189  /* predection based on system-model */
190  d_range_est = d_range_est - d_delta_t * d_velocity_est; // r_k = r_k-1 + delta_T * v_k-1 // v_k = v_k-1
191  // calc state-covariance matrix
192  P[0][0] = P[0][0]+P[1][0]*d_delta_t+d_delta_t*(P[0][1]+d_delta_t*P[1][1])+Q[0][0];
193  P[0][1] = P[0][1]+d_delta_t*P[1][1]+Q[0][1];
194  P[1][0] = P[1][0]+d_delta_t*P[1][1]+Q[1][0];
195  P[1][1] = P[1][1] + Q[1][1];
196 
197  /* filtering based on new measurement */
198  // calc Kalman-Gain: K = P*H'/(H*P*H'+R) here: K = P/(P+R)
199  a = P[0][0]+R[0][0];
200  b = P[0][1]+R[0][1];
201  c = P[1][0]+R[1][0];
202  d = P[1][1]+R[1][1];
203  factor = 1/(a*d-b*c); // for 2x2 matrix invers
204  K[0][0] = factor * (d*P[0][0] - c*P[0][1]);
205  K[0][1] = factor * (a*P[0][1] - b*P[0][0]);
206  K[1][0] = factor * (d*P[1][0] - c*P[1][1]);
207  K[1][1] = factor * (a*P[1][1] - b*P[1][0]);
208  // updated state
209  d_range_est = d_range_est + K[0][0]*(d_range_meas - d_range_est) + K[0][1]*(d_velocity_meas - d_velocity_est);
210  d_velocity_est = d_velocity_est + K[1][0]*(d_range_meas - d_range_est) + K[1][1]*(d_velocity_meas - d_velocity_est);
211  // update state-covariance matrix
212  P[0][0] = (1-K[0][0])*P[0][0] - P[1][0]*K[0][1];
213  P[0][1] = (1-K[0][0])*P[0][1] - P[1][1]*K[0][1];
214  P[1][0] = (1-K[1][1])*P[1][0] - P[0][0]*K[1][0];
215  P[1][1] = (1-K[1][1])*P[1][1] - P[0][1]*K[1][0];
216  } // end kalman-filter

218  void // begin particle filter
219  tracking_singletarget_impl::filter_particle(){
220  float lh, range_dif, velocity_dif;
221  float sum_weight, sum_weight_square;
222  sum_weight = 0;
223  sum_weight_square = 0;
224  for(int k=0; k<d_num_particle; k++){
225  // sample particles from time prosecution
226  d_particle_range[k] = random_normal(d_particle_range[k]-d_particle_velocity[k]*d_delta_t, std::sqrt(Q[0][0]));
227  d_particle_velocity[k] = random_normal(d_particle_velocity[k], std::sqrt(Q[1][1]));
228  range_dif = d_range_meas-d_particle_range[k];
229  velocity_dif = d_velocity_meas-d_particle_velocity[k];
230  lh = std::exp(-0.5*(range_dif*range_dif*R_inv[0][0]+velocity_dif*velocity_dif*R_inv[1][1]))/2/M_PI/std::sqrt(R_det); // get likelihood
231  d_particle_weight[k] = d_particle_weight[k]*lh;
232  // calc sum of weights and sum of weight square
233  sum_weight = sum_weight + d_particle_weight[k];
234  sum_weight_square = sum_weight_square + d_particle_weight[k]*d_particle_weight[k];
235  }
236  // normalize particle-weight ( sum(particle_weight =! 1))
237  for(int k=0; k<d_num_particle; k++) d_particle_weight[k] = d_particle_weight[k]/sum_weight;
238  float n_eff = 1/(float)sum_weight_square; // measure of degeneracy
239  if(n_eff<d_num_particle/(float)2){ // if degeneracy is to high --> systematic resampling
240  int num_range = 1000000;
241  float u1 = (rand()%num_range)/(float)num_range; // sample random starting index ...
242  u1 = u1/d_num_particle; // ... and normalize
243  std::vector<float> cum_weight;
244  cum_weight.resize(d_num_particle);
245  cum_weight[0] = d_particle_weight[0];
246  for(int k=1; k<d_num_particle; k++){ // calc cummulativ sum of all particles
247  cum_weight[k] = cum_weight[k-1]+d_particle_weight[k];
248  }
249  int i = 0;
250  std::vector<float> u;
251  u.resize(d_num_particle);
252  std::vector<int> index;
253  index.resize(d_num_particle);
254  for(int k=0; k<d_num_particle; k++){
255  u[k] = u1+k/(float)d_num_particle;
256  while(u[k]>cum_weight[i]){
257  i++;
258  }
259  index[k] = i; // chose particle indexes with high weight
260  }
261  for(int k=0; k<d_num_particle; k++){
262  d_particle_range[k] = d_particle_range[index[k]]; // resample only those particles with high weight ...
263  d_particle_velocity[k] = d_particle_velocity[index[k]];
264  d_particle_weight[k] = 1/(float)d_num_particle; // ... and set particle_weight to the same value
265  }
266  }
267 
268  d_range_est = 0;
269  d_velocity_est = 0;
270  // calc mean of all weighted particles to get the estimated state
271  for(int k=0; k<d_num_particle; k++){
272  d_range_est += d_particle_range[k]*d_particle_weight[k];
273  d_velocity_est += d_particle_velocity[k]*d_particle_weight[k];
274  }
275  } // end particle filter

277  bool
278  tracking_singletarget_impl::tracking(){
279  bool is_valid = not(d_is_empty); // measurement is valid if vectors are not empty
280 
281  // Calc time difference and matrix Q
282  d_delta_t = d_time-d_time_last; // time difference
283  d_time_last = d_time;
284 
285  Q[0][0] = 0.25*std::pow(d_delta_t,4)*d_std_accel_sys*d_std_accel_sys;
286  Q[0][1] = 0.5*std::pow(d_delta_t,3)*d_std_accel_sys*d_std_accel_sys;
287  Q[1][0] = 0.5*std::pow(d_delta_t,3)*d_std_accel_sys*d_std_accel_sys;
288  Q[1][1] = std::pow(d_delta_t,2)*d_std_accel_sys*d_std_accel_sys;
289 
290  // Do tracking
291  if(d_is_track){ // check if track is available
292  if(is_valid){ // check if measurement is valid
293  float range_dif, velocity_dif;
294  range_dif = d_range_meas-d_range_est;
295  velocity_dif = d_velocity_meas-d_velocity_est;
296  float lh = std::exp(-0.5*(range_dif*range_dif*R_inv[0][0]+velocity_dif*velocity_dif*R_inv[1][1]))/2/M_PI/std::sqrt(R_det); // calc likelihood
297 
298  if(d_threshold_track<lh){ // if new sample is accepted as track
299  if(d_filter == "particle") filter_particle();
300  else if(d_filter == "kalman") filter_kalman();
301  else throw std::runtime_error("No suitable filter found for given identifier.");
302  d_lost = 0;
303  }
304  else{ // if sample is rejected, do simple estimation based on system model (v=const, R=v*dt)
305  d_range_est = d_range_est-d_velocity_est*d_delta_t; // velocity < 0 for movement to radar; v=const.
306  d_lost++;
307  if(d_lost>d_threshold_lost){
308  d_is_track = false;
309  return false; // tracking is not successfull
310  }
311  }
312  }
313  else{ // if sample is rejected, do simple estimation based on system model (v=const, R=v*dt)
314  d_range_est = d_range_est-d_velocity_est*d_delta_t; // velocity < 0 for movement to radar; v=const.
315  d_lost++;
316  if(d_lost>d_threshold_lost){
317  d_is_track = false;
318  return false; // tracking is not successfull
319  }
320  }
321  }
322  else{ // if track is not available
323  if(is_valid){ // if measurement is valid, do initializiation; else do nothing
324  if(d_filter == "particle"){ // initialize particle filter
325  for(int k=0; k<d_num_particle; k++){
326  // samle particles from importance density based on first measurement
327  d_particle_range[k] = random_normal(d_range_meas, d_std_range_meas);
328  d_particle_velocity[k] = random_normal(d_velocity_meas, d_std_velocity_meas);
329  d_particle_weight[k] = 1/float(d_num_particle); // set all particle_weights to the same value
330  }
331  }
332  else if(d_filter == "kalman"){ // initialize state-covariance matrix for the kalman filter
333  P[0][0] = d_std_range_meas * d_std_range_meas;
334  P[0][1] = 0;
335  P[1][0] = 0;
336  P[1][1] = d_std_velocity_meas * d_std_velocity_meas;
337  }
338  else throw std::runtime_error("No suitable filter found for given identifier.");
339  // use first range & velocity measurement to intialize the state
340  d_range_est = d_range_meas;
341  d_velocity_est = d_velocity_meas;
342  d_is_track = true;
343  d_lost = 0;
344  }
345  else{
346  d_is_track = false;
347  d_lost = 0;
348  return false;
349  }
350  }
351  return true; // tracking is successfull
352  }
