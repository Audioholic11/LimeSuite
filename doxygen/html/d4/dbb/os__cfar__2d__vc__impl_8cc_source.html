91  int
92  os_cfar_2d_vc_impl::work (int noutput_items,
93  gr_vector_int &ninput_items,
94  gr_vector_const_void_star &input_items,
95  gr_vector_void_star &output_items)
96  {
97  const gr_complex *in = (const gr_complex *) input_items[0];
98 
99  // OS-CFAR detection (2D)
100 
101  // Get mean power
102  int ylen = ninput_items[0]; // size of y axis
103  float mean_power = 0;
104  for(int x=0; x<d_vlen; x++){ // go through x axis
105  for(int y=0; y<ylen; y++){ // go through y axis
106  mean_power += std::pow(std::abs(in[x+d_vlen*y]),2);
107  }
108  }
109  mean_power = mean_power/d_vlen/ylen;
110 
111  // Go through matrix and select CUT (cell under test)
112  d_x.clear();
113  d_y.clear();
114  d_power.clear();
115  std::vector<float> hold_samp; // vector for holding samples
116  for(int x=0; x<d_vlen; x++){ // go through x axis
117  for(int y=0; y<ylen; y++){ // go through y axis
118 
119  // Gather samples around CUT, logarithmic values are not possible!
120  hold_samp.clear();
121  for(int kx=1; kx<d_samp_compare[0]+1; kx++){ // go through compare samples on x axis
122  for(int ky=1; ky<d_samp_compare[1]+1; ky++){ // go through compare samples on y axis
123 
124  // Do not evaluate CUT
125  if(kx==0 && ky==0) continue;
126 
127  // Check lower right corner of the compare field
128  if(x+kx+d_samp_compare[0]>=d_vlen || y+ky+d_samp_compare[1]>=ylen){ // push back mean_power if not in matrix
129  hold_samp.push_back(mean_power);
130  }
131  else{ // push back value of the matrix
132  hold_samp.push_back(std::pow(std::abs(in[x+kx+d_samp_compare[0]+d_vlen*(y+ky+d_samp_compare[1])]),2));
133  }
134 
135  // Check upper right corner of the compare field
136  if(x+kx+d_samp_compare[0]>=d_vlen || y-ky-d_samp_compare[1]<0){ // push back mean_power if not in matrix
137  hold_samp.push_back(mean_power);
138  }
139  else{ // push back value of the matrix
140  hold_samp.push_back(std::pow(std::abs(in[x+kx+d_samp_compare[0]+d_vlen*(y-ky-d_samp_compare[1])]),2));
141  }
142 
143  // Check upper left corner of the compare field
144  if(x-kx-d_samp_compare[0]<0 || y-ky-d_samp_compare[1]<0){ // push back mean_power if not in matrix
145  hold_samp.push_back(mean_power);
146  }
147  else{ // push back value of the matrix
148  hold_samp.push_back(std::pow(std::abs(in[x-kx-d_samp_compare[0]+d_vlen*(y-ky-d_samp_compare[1])]),2));
149  }
150 
151  // Check lower left corner of the compare field
152  if(x-kx-d_samp_compare[0]<0 || y+ky+d_samp_compare[1]>=ylen){ // push back mean_power if not in matrix
153  hold_samp.push_back(mean_power);
154  }
155  else{ // push back value of the matrix as abs-square
156  hold_samp.push_back(std::pow(std::abs(in[x-kx-d_samp_compare[0]+d_vlen*(y+ky+d_samp_compare[1])]),2));
157  }
158 
159  }
160  }
161 
162  // Sort d_hold vector
163  std::sort(hold_samp.begin(),hold_samp.end());
164 
165  // Accept CUT if pow(abs(sort_vector[len_sort_vector*rel_threshold]*mult_threshold),2) < pow(abs(value_CUT),2)
166  float compare_value = hold_samp[int((hold_samp.size()-1)*d_rel_threshold)]*d_mult_threshold;
167  if(compare_value < std::pow(std::abs(in[x+d_vlen*y]),2)){
168  d_x.push_back(x);
169  d_y.push_back(y);
170  d_power.push_back(std::pow(std::abs(in[x+d_vlen*y]),2));
171  }
172 
173  }
174  }
175 
176  // Get rx_time tag
177  get_tags_in_range(d_tags,0,nitems_read(0),nitems_read(0)+1,pmt::string_to_symbol("rx_time"));
178 
179  // Setup output msg
180  if(d_tags.size()>0) d_ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),d_tags[0].value);
181  else d_ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),pmt::make_tuple(pmt::from_uint64(0),pmt::from_double(-1))); // if no timetag is found, set to 0 and frac_sec to -1
182  d_ppower = pmt::list2(pmt::string_to_symbol("power"),pmt::init_f32vector(d_power.size(),d_power));
183  d_px = pmt::list2(pmt::string_to_symbol("axis_x"),pmt::init_f32vector(d_x.size(),d_x));
184  d_py = pmt::list2(pmt::string_to_symbol("axis_y"),pmt::init_f32vector(d_y.size(),d_y));
185  d_value = pmt::list4(d_ptimestamp,d_px,d_py,d_ppower);
186 
187  // Publish message
188  message_port_pub(d_port_id,d_value);
189 
190  // Tell runtime system how many output items we produced.
191  return 0;
192  }
