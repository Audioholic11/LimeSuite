331

333  Reg_write(0x000A, regA);

345  return false;
346  }

348  {
349  UpdateStatus(LMS_TEST_INFO, " Failed to read samples: No samples received");
350  return false;
351  }
352 
353  Reg_write(0x000A, regA & ~0x3); //stop Tx Rx
354

358  int16_t* bufStart = (int16_t*)(data_buffer+2*sizeof(lime::FPGA_DataPacket)); // pointer buffer
359  //Parse samples from packet
360  for (int samplesCollected = 0; samplesCollected < fftSize; samplesCollected++)
361  {
362  if ((samplesCollected%1020) == 0) //skip packet headers
363  bufStart += 8;

365  m_fftCalcIn[samplesCollected].i = (float)*bufStart++;
366  }
367

369 
370  for (int i = 1; i<fftSize; ++i) //skip DC
371  {
372  float output = 20*log10(sqrt(m_fftCalcOut[i].r*m_fftCalcOut[i].r + m_fftCalcOut[i].i*m_fftCalcOut[i].i)/(fftSize*(1<<15)));
373  if (output > peakAmplitude)
374  {
375  peakAmplitude = output;
376  peakFrequency = i * samplerate / fftSize;
377  }
378  }
379 
380  if (peakFrequency > samplerate/2)
381  peakFrequency = peakFrequency-samplerate;
382 
383  if ((peakAmplitude > peakval - rfTestTolerance_dB)
384  && (peakAmplitude < peakval + rfTestTolerance_dB)
385  && (peakFrequency < peakfreq+ rfTestTolerance_Hz)
386  && (peakFrequency > peakfreq- rfTestTolerance_Hz))
387  {
388  passed = true;
389  }
390 
391  free(m_kissBuffer);
392  delete[] m_fftCalcOut;
393  delete[] m_fftCalcIn;
