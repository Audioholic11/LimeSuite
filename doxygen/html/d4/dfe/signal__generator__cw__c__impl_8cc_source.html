66  signal_generator_cw_c_impl::work(int noutput_items,
67  gr_vector_const_void_star &input_items,
68  gr_vector_void_star &output_items)
69  {
70  gr_complex *out = (gr_complex *) output_items[0];
71 
72  // Set output to zero
73  std::memset(out, 0, noutput_items*sizeof(gr_complex));
74 
75  // Integrate phase for iq signal
76  for(int i=0; i<noutput_items; i++){
77  // Set tag on every packet_len-th item
78  if((nitems_written(0)+i)%d_packet_len==0) add_item_tag(0, nitems_written(0)+i, d_key, d_value, d_srcid);
79 
80  // Write sample
81  for(int k=0; k<d_num_freq; k++){ // Go through frequencies
82  out[i] += d_amplitude/(float)d_num_freq*exp(d_phase[k]); // output amplitude is normed on d_amplitude
83  d_phase[k] = 1j*std::fmod(imag(d_phase[k])+2*M_PI*d_frequency[k]/(float)d_samp_rate,2*M_PI);
84  }
85  }
86 
87  // Tell runtime system how many output items we produced.
