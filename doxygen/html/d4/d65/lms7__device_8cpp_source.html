826  }

831  div = (2<<(ratio));

863  }
864  else
865  {
866  gfir[i*8+j] = 0;
867  }
868  }
869  }
870 
871  div -= 1;
872  L = div > 7 ? 7 : div;
873 
874  if (tx)
875  {
876  if (filt ==LMS_GFIR1)
877  {
878  if ((lms->Modify_SPI_Reg_bits(LMS7param(GFIR1_L_TXTSP), L) != 0)
879  || (lms->Modify_SPI_Reg_bits(LMS7param(GFIR1_N_TXTSP), div) != 0))

883  {
884  if ((lms->Modify_SPI_Reg_bits(LMS7param(GFIR2_L_TXTSP), L) != 0)

886  return -1;
887  }
888  else if (filt ==LMS_GFIR3)
889  {
890  if ((lms->Modify_SPI_Reg_bits(LMS7param(GFIR3_L_TXTSP), L) != 0)

892  return -1;
893  }
894  }
