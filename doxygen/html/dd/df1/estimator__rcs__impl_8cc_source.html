138  float
139  estimator_rcs_impl::calculate_rcs() {
140  // catch errors
141  if(d_range.size() == 0) throw std::runtime_error("range vector has size zero");
142  if(d_power.size() == 0) throw std::runtime_error("power vector has size zero");
143 
144  // regard usrp gain and signal path
145  float power_rx = pow(d_power[0], d_exponent) / d_power_tx / pow(10, d_usrp_gain_rx/10);
146 
147  float fak = d_fak * pow(d_range[0], 4);
148 
149  // debug output
150  // std::cout << "PowerTx: " << d_power_tx << std::endl;
151  // std::cout << "PowerRx: " << power_rx << std::endl;
152  // std::cout << "Lambda: " << d_lambda << std::endl;
153  // std::cout << "GainRx: " << d_antenna_gain_rx << std::endl;
154  // std::cout << "GainTx: " << d_antenna_gain_tx << std::endl;
155  // std::cout << "fak: " << fak << std::endl;
156 
157  return power_rx/d_power_tx * fak * d_corr_factor;
158  }

160  void
161  estimator_rcs_impl::handle_msg(pmt::pmt_t msg)
162  {
163  // Read msg from peak detector
164  d_msg_hold.clear();
165  pmt::pmt_t msg_part;
166  bool found_range = false;
167  bool found_power = false;
168  for(int k=0; k<pmt::length(msg); k++){
169  msg_part = pmt::nth(k,msg);
170  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="range"){
171  d_prange = pmt::nth(1,msg_part);
172  d_msg_hold.push_back(msg_part);
173  found_range = true;
174  }
175  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="power"){
176  d_ppower = pmt::nth(1,msg_part);
177  found_power = true;
178  }
179  else{
180  d_msg_hold.push_back(msg_part);
181  }
182  }
183 
184  if(not(found_range&&found_power)) throw std::runtime_error("range or power identifier (symbol) not found");
185 
186  d_range = pmt::f32vector_elements(d_prange);
187  d_power = pmt::f32vector_elements(d_ppower);
188  d_rcs.clear();
189 
190  if(d_range.size()!=d_power.size()) throw std::runtime_error("range and power vectors do not have same size");
191 
192  // Calculate RCS
193  float rcs_mean = 0.0;
194  if(d_range.size() == 0 && d_power.size() == 0){
195  //std::cout << "ERROR: No target detected for RCS calculation" << std::endl;
196  }
197  else {
198  d_rcs_vals.push_back(calculate_rcs());
199  }
200  //std::cout << "RCS_TEMP: " << calculate_rcs() << std::endl;
201 
202  if(d_loop_counter+1 >= d_num_mean) {
203  rcs_mean = calculate_vector_mean(&d_rcs_vals);
204  }
205  else {
206  d_loop_counter++;
207  }
208 
209  for(int k=0; k<d_range.size(); k++){
210  d_rcs.push_back(rcs_mean);
211  }
212 
213  // Push pmt to output msg port
214  d_rcs_key = pmt::string_to_symbol("rcs"); // identifier velocity
215  d_rcs_value = pmt::init_f32vector(d_rcs.size(), d_rcs); // vector to pmt
216  d_rcs_pack = pmt::list2(d_rcs_key, d_rcs_value); // make list for velocity information
217 
218  d_value = pmt::list1(d_rcs_pack);
219  for(int k=0; k<d_msg_hold.size(); k++){
220  d_value = pmt::list_add(d_value, d_msg_hold[k]);
221  }
222 
223  message_port_pub(d_port_id_out,d_value); // publish message
224  }
