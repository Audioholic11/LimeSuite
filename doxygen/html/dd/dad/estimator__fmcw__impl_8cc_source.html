123  void
124  estimator_fmcw_impl::estimate()
125  {
126  // Get timestamp and frequencies (cw, up-chirp, down-chirp)
127  std::vector<float> freq_cw, freq_up, freq_down;
128  pmt::pmt_t timestamp;
129  pmt::pmt_t msg_part;
130  pmt::pmt_t power;
131 
132  for(int k=0; k<pmt::length(d_msg_cw); k++){ // search freq CW
133  msg_part = pmt::nth(k,d_msg_cw);
134  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="frequency"){
135  freq_cw = pmt::f32vector_elements(pmt::nth(1,msg_part));
136  }
137  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="rx_time"){
138  timestamp = pmt::nth(1,msg_part);
139  }
140  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="power"){
141  power = msg_part;
142  }
143  }
144 
145  for(int k=0; k<pmt::length(d_msg_up); k++){ // search freq UP
146  msg_part = pmt::nth(k,d_msg_up);
147  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="frequency"){
148  freq_up = pmt::f32vector_elements(pmt::nth(1,msg_part));
149  }
150  }
151 
152  for(int k=0; k<pmt::length(d_msg_down); k++){ // search freq DOWN
153  msg_part = pmt::nth(k,d_msg_down);
154  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="frequency"){
155  freq_down = pmt::f32vector_elements(pmt::nth(1,msg_part));
156  }
157  }
158 
159  // Get velocities out of CW frequencies
160  std::vector<float> velocity_cw;
161  for(int k=0; k<freq_cw.size(); k++){
162  velocity_cw.push_back(-c_light/2/d_center_freq*freq_cw[k]); // with minus! for same sign as all_velocities
163  }
164 
165  // Get all possible range/velocity pairs
166  std::vector<float> all_ranges, all_velocities;
167  float v1, v2, r;
168  for(int m=0; m<freq_up.size(); m++){
169  for(int n=0; n<freq_down.size(); n++){
170  r = (freq_up[m]-freq_down[n])/(d_const_up+d_const_down); // range from up- and down-chirp
171  all_ranges.push_back(r);
172  v1 = (d_const_up*r-freq_up[m])/d_const_doppler; // velocity from up-chirp
173  v2 = (-d_const_down*r-freq_down[n])/d_const_doppler; // velocity from down-chirp
174  all_velocities.push_back((v1+v2)/2.0); // median of up- and down-chirp
175  }
176  }
177 
178  // Minimize velocity from CW with all possible velocities from up-/down-chirp
179  std::vector<float> velocity, range;
180  int min_vel_index;
181  float min_vel;
182  for(int k=0; k<velocity_cw.size(); k++){
183  min_vel = 1e20;
184  min_vel_index = -1;
185 
186  for(int l=0; l<all_velocities.size(); l++){
187  if(min_vel>std::abs(velocity_cw[k]-all_velocities[l])){
188  min_vel = std::abs(velocity_cw[k]-all_velocities[l]);
189  min_vel_index = l;
190  }
191  }
192 
193  velocity.push_back(velocity_cw[k]); // push back cw velocity because of (probably) better resolution
194  range.push_back(all_ranges[min_vel_index]);
195  }
196 
197  // Pack output msg and push to output
198  pmt::pmt_t time_pack;
199  time_pack = pmt::list2(pmt::string_to_symbol("rx_time"), timestamp); // make list for timestamp information
200 
201  pmt::pmt_t vel_value, vel_pack;
202  vel_value = pmt::init_f32vector(velocity.size(), velocity); // vector to pmt
203  vel_pack = pmt::list2(pmt::string_to_symbol("velocity"), vel_value); // make list for velocity information
204 
205  pmt::pmt_t range_value, range_pack;
206  range_value = pmt::init_f32vector(range.size(), range); // vector to pmt
207  range_pack = pmt::list2(pmt::string_to_symbol("range"), range_value); // make list for range information
208 
209  pmt::pmt_t value;
210  if(d_push_power){ // if power of peaks shall be pushed through
211  value = pmt::list4(time_pack, vel_pack, range_pack, power); // all information to one pmt list
212  }
213  else{
214  value = pmt::list3(time_pack, vel_pack, range_pack); // all information to one pmt list
215  }
216  message_port_pub(d_port_id_out,value);
217  }
