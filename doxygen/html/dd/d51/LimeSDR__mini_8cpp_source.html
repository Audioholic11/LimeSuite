106 int LMS7_LimeSDR_mini::SetFrequency(bool isTx, unsigned chan, double f_Hz)
107 {
108  lime::LMS7002M* lms = lms_list[0];
109 
110  ChannelInfo& channel = isTx ? tx_channels[0] : rx_channels[0];
111  channel.freq = f_Hz;
112 
113  auto setTDD = [=](double center)->int
114  {
115  ChannelInfo& other = isTx ? rx_channels[0] : tx_channels[0];
116  bool tdd = fabs(other.freq+other.cF_offset_nco-center) > 0.1 ? false : true;
117  lms->EnableSXTDD(tdd);
118  if (isTx || (!tdd))
119  if (lms->SetFrequencySX(isTx, center) != 0)
120  return -1;
121  return 0;
122  };
123 
124  if (f_Hz < 30e6)
125  {
126  if (setTDD(30e6) != 0)
127  return -1;
128  channel.cF_offset_nco = 30e6-f_Hz;
129  if (SetRate(isTx,GetRate(isTx,0),2)!=0)
130  return -1;
131  return 0;
132  }
133 
134  if (channel.cF_offset_nco != 0)
135  SetNCOFreq(isTx, 0, -1, 0.0);
136  channel.cF_offset_nco = 0;
137  if (setTDD(f_Hz) != 0)
138  return -1;
139  return 0;
140 }

150 int LMS7_LimeSDR_mini::SetPath(bool tx, unsigned chan, unsigned path)
151 {
152  lime::LMS7002M* lms = lms_list[0];
153  if (lms->Modify_SPI_Reg_bits(LMS7param(MAC), (chan%2) + 1) != 0)
154  return -1;
155  if (tx==false)
156  {
157  if ((lms->Modify_SPI_Reg_bits(LMS7param(SEL_PATH_RFE),path)!=0)
158  || (lms->Modify_SPI_Reg_bits(LMS7param(EN_INSHSW_L_RFE), path != 2) != 0)
159  || (lms->Modify_SPI_Reg_bits(LMS7param(EN_INSHSW_W_RFE), path != 3) != 0))
160  return -1;
161  if (path==LMS_PATH_LNAW)
162  {
163  uint16_t value;
164  connection->ReadRegister(0x17,value);
165  value &= ~(1<<8);
166  value |= 1<<9;
167  connection->WriteRegister(0x17, value);
168  }
169  else if (path==LMS_PATH_LNAH)
170  {
171  uint16_t value;
172  connection->ReadRegister(0x17,value);
173  value &= ~(1<<9);
174  value |= 1<<8;
175  connection->WriteRegister(0x17, value);
176  }
177  else if (LMS_PATH_LNAL)
178  lime::warning("LNAL has no connection to RF ports");
179  }
180  else
181  {
182  if ((lms->Modify_SPI_Reg_bits(LMS7param(SEL_BAND1_TRF), path == LMS_PATH_TX1) != 0)
183  || (lms->Modify_SPI_Reg_bits(LMS7param(SEL_BAND2_TRF), path == LMS_PATH_TX2) != 0))
184  return -1;
185  if (path==LMS_PATH_TX1)
186  {
187  uint16_t value;
188  connection->ReadRegister(0x17,value);
189  value &= ~(1<<13);
190  value |= 1<<12;
191  connection->WriteRegister(0x17, value);
192  }
193  else if (path==LMS_PATH_TX2)
194  {
195  uint16_t value;
196  connection->ReadRegister(0x17,value);
197  value &= ~(1<<12);
198  value |= 1<<13;
199  connection->WriteRegister(0x17, value);
200  }
201  }
202  return 0;
203 }

205 int LMS7_LimeSDR_mini::SetRate(double f_Hz, int oversample)
206 {
207  lime::LMS7002M* lms = lms_list[0];
208 
209  if (oversample == 0)
210  oversample = lime::cgenMax/(16*f_Hz);
211  bool bypass = (oversample <= 1 && tx_channels[0].cF_offset_nco == 0.0 && rx_channels[0].cF_offset_nco == 0.0);
212 
213  if ((lms->Modify_SPI_Reg_bits(LMS7_LML1_SISODDR, 1)!=0)
214  || (lms->Modify_SPI_Reg_bits(LMS7_LML2_SISODDR, 1)!=0)
215  || (lms->Modify_SPI_Reg_bits(LMS7_CDSN_RXALML, !bypass)!=0))
216  return -1;
217 
218  if (!bypass)
219  return LMS7_Device::SetRate(f_Hz, oversample);
220 
221  tx_channels[0].sample_rate = f_Hz;
222  rx_channels[0].sample_rate = f_Hz;
223 
224  if ((lms->SetFrequencyCGEN(f_Hz*4) != 0)
225  || (lms->Modify_SPI_Reg_bits(LMS7param(EN_ADCCLKH_CLKGN), 0) != 0)
226  || (lms->Modify_SPI_Reg_bits(LMS7param(CLKH_OV_CLKL_CGEN), 2) != 0)
227  || (lms->Modify_SPI_Reg_bits(LMS7param(MAC), 1) != 0)
228  || (lms->SetInterfaceFrequency(lms->GetFrequencyCGEN(), 7, 7) != 0))
229  return -1;
230 
231  double fpgaTxPLL = lms->GetReferenceClk_TSP(lime::LMS7002M::Tx);
232  double fpgaRxPLL = lms->GetReferenceClk_TSP(lime::LMS7002M::Rx);
233  if (fpga->SetInterfaceFreq(fpgaTxPLL, fpgaRxPLL, 0) != 0)
234  return -1;
235  return 0;
236 }
