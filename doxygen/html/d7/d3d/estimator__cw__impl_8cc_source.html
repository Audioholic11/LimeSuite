66  estimator_cw_impl::handle_msg(pmt::pmt_t msg)
67  {
68  // Read msg from peak detector
69  pmt::pmt_t msg_part;
70  for(int k=0; k<pmt::length(msg); k++){
71  msg_part = pmt::nth(k,msg);
72  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="frequency"){
73  d_pfreq = pmt::nth(1,msg_part);
74  }
75  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="rx_time"){
76  d_ptimestamp = pmt::nth(1,msg_part);
77  }
78  }
79 
80  d_freq = pmt::f32vector_elements(d_pfreq); // get freq vector
81 
82  // Calc velocities and write to vector
83  d_vel.clear();
84  for(int k=0; k<d_freq.size(); k++){
85  d_vel.push_back(d_freq[k]*c_light/2/d_center_freq); // calc with doppler formula
86  }
87 
88  // Push pmt to output msg port
89  d_vel_key = pmt::string_to_symbol("velocity"); // identifier velocity
90  d_vel_value = pmt::init_f32vector(d_vel.size(), d_vel); // vector to pmt
91  d_vel_pack = pmt::list2(d_vel_key, d_vel_value); // make list for velocity information
92 
93  d_time_key = pmt::string_to_symbol("rx_time"); // identifier timestamp
94  d_time_pack = pmt::list2(d_time_key, d_ptimestamp); // make list for timestamp (rx_time) information
95 
96  d_value = pmt::list2(d_time_pack, d_vel_pack); // all information to one pmt list (only velocity -> only 1 item list)
97  message_port_pub(d_port_id_out,d_value);
98  }
