61  trigger_command_impl::handle_msg(pmt::pmt_t msg){
62  // Check if execution has to be blocked
63  d_actual_time = boost::posix_time::microsec_clock::local_time();
64  d_time_duration = d_actual_time-d_last_time;
65  if(d_time_duration.total_milliseconds()>d_block_time){
66  d_last_time = boost::posix_time::microsec_clock::local_time();
67  }
68  else{
69  return;
70  }
71 
72  // Go through msg parts and check symbols
73  pmt::pmt_t msg_part_symbol, msg_part_value;
74  bool execute_command = true;
75  for(int k=0; k<pmt::length(msg); k++){
76  msg_part_symbol = pmt::nth(0,pmt::nth(k,msg));
77  msg_part_value = pmt::nth(1,pmt::nth(k,msg));
78  // Go through given symbols
79  for(int p=0; p<d_identifiers.size(); p++){
80  if(pmt::symbol_to_string(msg_part_symbol)==d_identifiers[p]){
81  // Go through input if values are a f32vector, else set executing command on false
82  if(pmt::is_f32vector(msg_part_value)){
83  // Go through values and check boundries
84  std::vector<float> values = pmt::f32vector_elements(msg_part_value);
85  for(int m=0; m<values.size(); m++){
86  if( values[m]<d_vals_min[p] || values[m]>d_vals_max[p] ){
87  execute_command = false;
88  }
89  }
90  }
91  // add here more types as else if
92  else{
93  execute_command = false;
94  }
95  }
96  }
97  }
98 
99  // Execute command if values are correct
100  if(execute_command){
101  int sys_return = std::system(d_command.c_str());
102  }
103  }
