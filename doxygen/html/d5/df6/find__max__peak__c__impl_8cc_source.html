88  int
89  find_max_peak_c_impl::work (int noutput_items,
90  gr_vector_int &ninput_items,
91  gr_vector_const_void_star &input_items,
92  gr_vector_void_star &output_items)
93  {
94  const gr_complex *in = (const gr_complex *) input_items[0];
95 
96  // Find max peak detection
97  d_freq.clear();
98  d_pks.clear();
99  d_angle.clear();
100 
101  int k_max_pos, k_max_neg;
102  if(d_cut_max_freq){
103  k_max_pos = (ninput_items[0]/float(d_samp_rate))*d_max_freq[1];
104  k_max_neg = ninput_items[0]+(ninput_items[0]/float(d_samp_rate))*d_max_freq[0];
105  }
106  int k = -1;
107  float hold = -1;
108  for(int p=d_samp_protect; p<ninput_items[0]-d_samp_protect; p++){ // implementation of protected samples
109  if(d_cut_max_freq){
110  if( p>k_max_pos && p<k_max_neg ){ // skip frequency over abs(max_freq)
111  continue;
112  }
113  }
114  if(std::pow(std::abs(in[p]),2)>hold && std::pow(std::abs(in[p]),2)>std::pow(10,d_threshold/10.0)){
115  hold = std::pow(std::abs(in[p]),2);
116  k = p;
117  }
118  }
119 
120  if(k!=-1){
121  if(k<=ninput_items[0]/2) d_freq.push_back(k*(d_samp_rate/(float)ninput_items[0])); // add frequency to message vector d_freq
122  else d_freq.push_back(-(float)d_samp_rate+k*(d_samp_rate/(float)ninput_items[0]));
123  d_pks.push_back(std::pow(std::abs(in[k]),2));
124  d_angle.push_back(std::arg(in[k]));
125  }
126 
127  // get rx_time tag
128  get_tags_in_range(d_tags,0,nitems_read(0),nitems_read(0)+1,pmt::string_to_symbol("rx_time"));
129 
130  // setup msg pmt
131  if(d_tags.size()>0) d_ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),d_tags[0].value);
132  else d_ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),pmt::make_tuple(pmt::from_uint64(0),pmt::from_double(-1))); // if no timetag is found, set to 0 and frac_sec to -1
133  d_pfreq = pmt::list2(pmt::string_to_symbol("frequency"),pmt::init_f32vector(d_freq.size(),d_freq));
134  d_ppks = pmt::list2(pmt::string_to_symbol("power"),pmt::init_f32vector(d_pks.size(),d_pks));
135  d_pangle = pmt::list2(pmt::string_to_symbol("phase"),pmt::init_f32vector(d_angle.size(),d_angle));
136  d_value = pmt::list4(d_ptimestamp,d_pfreq,d_ppks,d_pangle);
137 
138  // publish message
139  message_port_pub(d_port_id,d_value);
140 
141  // Tell runtime system how many output items we produced.
142  return 0;
143  }
