153 int ReadWFM(const wxString filename, std::vector<int16_t> &iSamples, std::vector<int16_t> &qSamples)
154 {
155  if (filename.length() <= 1)
156  return 0;
157  wxFFile fpin(filename, "rb");
158  unsigned char inputBuffer[4];
159  unsigned char c1, c2, c3, c4; // To read Agilent Signal Studio file
160  double iin, qin; // IQ inputs
161  int iint, qint; // IQ integer versions
162  int cnt = 0;
163 
164  if( fpin.IsOpened() == false)
165  {
166 #ifndef NDEBUG
167  cout << _("WFM reading: Input file can not be opened (") << filename << _(")\n");
168 #endif
169  return -1;
170  }
171 
172  while( fpin.Read(inputBuffer, 4) == 4 )
173  {
174  c1 = inputBuffer[0];
175  c2 = inputBuffer[1];
176  c3 = inputBuffer[2];
177  c4 = inputBuffer[3];
178 
179  cnt++;
180 
181  c1 &= 0xFF;
182  c2 &= 0xFF;
183  c3 &= 0xFF;
184  c4 &= 0xFF;
185 
186  if( c1&0x80 ) iin = (double)(-1*(1<<15) + ((c1&0x7F)<<8) + c2);
187  else iin = (double)(((c1&0x7F)<<8) + c2);
188 
189  if( c3&0x80 ) qin = (double)(-1*(1<<15) + ((c3&0x7F)<<8) + c4);
190  else qin = (double)(((c3&0x7F)<<8) + c4);
191 
192  iint = (int)(iin);
193  qint = (int)(qin);
194 
195  //convert from 16 bit to 12 bit values
196  //iint = iint >> 4;
197  //qint = qint >> 4;
198 
199  iSamples.push_back(iint);
200  qSamples.push_back(qint);
201  };
202  return iSamples.size();
203 }

249 int FPGAcontrols_wxgui::UploadFile(std::vector<int16_t> isamples, std::vector<int16_t> qsamples)
250 {
251  if (!lmsControl)
252  {
253  wxMessageBox(_("Device not connected"), _("Error"));
254  return -2;
255  }
256 
257  progressBar->SetRange(isamples.size());
258  progressBar->SetValue(0);
259 
260  btnPlayWFM->Enable(false);
261  btnStopWFM->Enable(false);
262 
263  const uint8_t chCount = 2;
264  bool MIMO = chkMIMO->IsChecked();
265 
266  lime::complex16_t* src[chCount];
267  for(int i=0; i<chCount; ++i)
268  src[i] = new lime::complex16_t[isamples.size()];
269 
270  for(size_t i=0; i<isamples.size(); ++i)
271  {
272  for(int c=0; c<chCount; ++c)
273  {
274  if(c == 1 && !MIMO)
275  {
276  src[c][i].i = 0;
277  src[c][i].q = 0;
278  }
279  else
280  {
281  src[c][i].i = isamples[i];
282  src[c][i].q = qsamples[i];
283  }
284  }
285  }
286 
287  int status = LMS_UploadWFM(lmsControl, (const void**)src, 2+cmbDevice->GetSelection()*2, isamples.size(), 1);
288 
289  progressBar->SetValue(progressBar->GetRange());
290  lblProgressPercent->SetLabelText(_("100%"));
291 
292  LMS_EnableTxWFM(lmsControl, cmbDevice->GetSelection()*2, true);
293 
294  btnPlayWFM->Enable(true);
295  btnStopWFM->Enable(true);
296 
297  if (status != 0)
298  {
299  wxMessageBox(_("Failed to upload WFM file"), _("Error"));
300  return -3;
301  }
302  else
303  return 0;
