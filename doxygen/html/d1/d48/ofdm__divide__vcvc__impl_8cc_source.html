74  int
75  ofdm_divide_vcvc_impl::work (int noutput_items,
76  gr_vector_int &ninput_items,
77  gr_vector_const_void_star &input_items,
78  gr_vector_void_star &output_items)
79  {
80  const gr_complex *in0 = (const gr_complex *) input_items[0];
81  const gr_complex *in1 = (const gr_complex *) input_items[1];
82  gr_complex *out = (gr_complex *) output_items[0];
83 
84  // Set noutput_items
85  noutput_items = ninput_items[0];
86 
87  // Set output buffer to zero -> is zeropadding
88  std::memset(out,0,sizeof(gr_complex)*noutput_items*d_vlen_out);
89 
90  // Do division and keep spaces between packets if vlen_out>vlen_in
91  // If actual vector is a sync words (given with num_sync_words) do not apply discarded carriers rule
92 
93  int next_discarded_element = 0; // set first discarded element on first vector item
94  if(d_discarded_carriers.size()==0){ // set first discarded element on first vector item
95  d_discarded_carriers.resize(1);
96  d_discarded_carriers[0] = d_vlen_in; // this disables discarded carriers
97  }
98 
99  // Divide items and discard carriers
100  for(int k=0; k<noutput_items; k++){
101  for(int l=0; l<d_vlen_in; l++){
102  if(k<d_num_sync_words){ // if actual vector is a sync word
103  out[k*d_vlen_out+l] = (in0[k*d_vlen_in+l])/(in1[k*d_vlen_in+l]);
104  }
105  else{ // if actual vector is NOT a sync word
106  if(l==d_discarded_carriers[next_discarded_element]){ // if actual element shall be discarded and set to zero
107  out[k*d_vlen_out+l] = 0;
108  if(next_discarded_element<d_discarded_carriers.size()-1) next_discarded_element++; // set next discarded element on next vector item
109  else next_discarded_element=0; // if item is last one jump back to first item in vector
110  }
111  else{ // if actual element shall be divided
112  out[k*d_vlen_out+l] = (in0[k*d_vlen_in+l])/(in1[k*d_vlen_in+l]);
113  }
114  }
115  }
116  }
117 
118  // Tell runtime system how many output items we produced.
119  return noutput_items;
120  }
