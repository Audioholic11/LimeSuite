232  void parse(std::istream& file)
233  {
234  fini_char_t line[FINI_BUFFER_SIZE];
235  bool first = true;
236  fini_sstream_t out;
237 
238  while(!file.eof())
239  {
240  file.getline(line, FINI_BUFFER_SIZE);
241 
242  if (first)
243  {
244  first = false;
245  if (line[0] == 0xEF) //Allows handling of UTF-16/32 documents
246  {
247  memmove(line, line + (CHAR_SIZE * 3), CHAR_SIZE * (FINI_BUFFER_SIZE - 3));
248  return;
249  }
250  }
251 
252  nake(line);
253 
254  if (line[0])
255  {
256  size_t len = fini_strlen(line);
257  if (len > 0 && !((len >= 2 && (line[0] == '/' && line[1] == '/')) || (len >= 1 && line[0] == '#'))) //Ignore comment and empty lines
258  {
259  if (line[0] == '[') //Section
260  {
261  section_t section;
262  size_t length = fini_strlen(line) - 2; //Without section brackets
263  while(isspace(line[length + 1])) //Leave out any additional new line characters, not "spaces" as the name suggests
264  --length;
265 
266  fini_char_t* ssection = (fini_char_t*)calloc(CHAR_SIZE, length + 1);
267  fini_strncpy(ssection, line + 1, length); //Count after first bracket
268 
269  current = new keys_t;
270 
271  out << ssection;
272  free(ssection);
273  Converters::GetLine(out, section);
274 
275  sections[section] = current;
276  }
277  else //Key
278  {
279  key_t key;
280  value_t value;
281 
282  fini_char_t* skey;
283  fini_char_t* svalue;
284 
285  skey = fini_strtok(line, _T("="));
286  svalue = fini_strtok(NULL, _T("\n"));
287 
288  if (skey && svalue)
289  {
290  size_t index = 0; //Without section brackets
291  while(isspace(skey[index])) //Leave out any additional new line characters, not "spaces" as the name suggests
292  index++;
293 
294  if (index != 0) //Has preceeding white space
295  fini_strcpy(skey, skey + index);
296 
297  out << skey;
298 
299  Converters::GetLine(out, key);
300 
301  out.clear();
302  out.str(fini_string_t());
303 
304  out << svalue;
305  Converters::GetLine(out, value);
306 
307  if (value != value_t())
308  (*current)[key] = value;
309  }
310  }
311 
312  out.clear();
313  out.str(fini_string_t()); //Clear existing stream;
314  }
315  }
316  }
