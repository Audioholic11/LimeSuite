838 void SoapyLMS7::writeSetting(const int direction, const size_t channel, const std::string &key, const std::string &value)
839 {
840  std::unique_lock<std::recursive_mutex> lock(_accessMutex);
841  const bool isTx = (direction == SOAPY_SDR_TX);
842 
843  if (key == "TSP_CONST")
844  {
845  const auto ampl = std::stoi(value);
846  lms7Device->SetTestSignal(isTx, channel, LMS_TESTSIG_DC, ampl, 0);
847  }
848 
849  else if (key == "CALIBRATE_TX" or (isTx and key == "CALIBRATE"))
850  {
851  double bw = std::stof(value);
852  SoapySDR::logf(SOAPY_SDR_INFO, "Calibrate Tx %f", bw);
853  if (lms7Device->Calibrate(true, channel, bw, 0)!=0)
854  throw std::runtime_error(lime::GetLastErrorMessage());
855  _channelsToCal.erase(std::make_pair(direction, channel));
856  }
857 
858  else if (key == "CALIBRATE_RX" or (not isTx and key == "CALIBRATE"))
859  {
860  double bw = std::stof(value);
861  SoapySDR::logf(SOAPY_SDR_INFO, "CalibrateRx %f", bw);
862  if (lms7Device->Calibrate(false, channel, bw, 0)!=0)
863  throw std::runtime_error(lime::GetLastErrorMessage());
864  _channelsToCal.erase(std::make_pair(direction, channel));
865  }
866 
867  else if (key == "ENABLE_GFIR_LPF")
868  {
869  double bw = std::stof(value);
870  SoapySDR::logf(SOAPY_SDR_INFO, "Configurate GFIR LPF %f", bw);
871  lms7Device->ConfigureGFIR(isTx, channel, true, bw);
872  }
873 
874  else if (key == "DISABLE_GFIR_LPF")
875  {
876  SoapySDR::logf(SOAPY_SDR_INFO, "Disable GFIR LPF");
877  lms7Device->ConfigureGFIR(isTx, channel, false, 0.0);
878  }
879 
880  else if (key == "TSG_NCO")
881  {
882  auto select = std::stoi(value);
883  if (select == -1)
884  {
885  //Requested to disable the TSG
886  lms7Device->SetTestSignal(isTx, channel, LMS_TESTSIG_NONE);
887  }
888  else if(select == 4)
889  {
890  lms7Device->SetTestSignal(isTx, channel, LMS_TESTSIG_NCODIV4F);
891  }
892  else if (select == 8)
893  {
894  lms7Device->SetTestSignal(isTx, channel, LMS_TESTSIG_NCODIV8F);
895  }
896  else
897  {
898  throw std::runtime_error("Invalid TSG_NCO option: " + value);
899  }
900  }
901  else
902  {
903  uint16_t val = std::stoi(value);
904  if (lms7Device->WriteParam(key,val,channel)!=-1)
905  return;
906  throw std::runtime_error("unknown setting key: "+key);
907  }
908 }
