70  void
71  msg_manipulator_impl::handle_msg(pmt::pmt_t msg){
72  // Go through msg parts and check symbols
73  pmt::pmt_t msg_part_symbol, msg_part_value;
74  bool symbol_found;
75  std::vector<float> value_new;
76  std::vector<pmt::pmt_t> store_msg;
77  int index;
78  for(int k=0; k<pmt::length(msg); k++){
79  symbol_found = false;
80  msg_part_symbol = pmt::nth(0,pmt::nth(k,msg));
81  msg_part_value = pmt::nth(1,pmt::nth(k,msg));
82  // Go through given symbols
83  for(int p=0; p<d_symbols.size(); p++){
84  if(pmt::symbol_to_string(msg_part_symbol)==d_symbols[p]){
85  index = p;
86  symbol_found = true;
87  }
88  }
89  // Store msg part
90  if(symbol_found && pmt::is_f32vector(msg_part_value)){ // use const_add and const_mult if symbol found and value is f32 vector, then store new msg
91  value_new = pmt::f32vector_elements(msg_part_value);
92  for(int m=0; m<value_new.size(); m++){ // manipulate vector
93  value_new[m] = (value_new[m]+d_const_add[index])*d_const_mult[index];
94  }
95  store_msg.push_back(pmt::list2(msg_part_symbol,pmt::init_f32vector(value_new.size(), value_new)));
96  }
97  else{ // just store msg if not found
98  store_msg.push_back(pmt::nth(k,msg));
99  }
100  }
101 
102  // Repack msg
103  pmt::pmt_t msg_out;
104  if(store_msg.size()==0) throw std::runtime_error("Store msg vector has unvalid length (size=0)");
105  msg_out = pmt::list1(store_msg[0]); // init pmt list
106  for(int k=1; k<store_msg.size(); k++){ // add pmt list
107  msg_out = pmt::list_add(msg_out, store_msg[k]);
108  }
109 
110  // Publish msg
111  message_port_pub(d_port_id_out,msg_out);
112  }
