85  ss << " [key=" << argInfo.key;
86  if (not argInfo.units.empty()) ss << ", units=" << argInfo.units;
87  if (not argInfo.value.empty()) ss << ", default=" << argInfo.value;
88 
89  //type
90  switch (argInfo.type)
91  {
92  case SoapySDR::ArgInfo::BOOL: ss << ", type=bool"; break;
93  case SoapySDR::ArgInfo::INT: ss << ", type=int"; break;
94  case SoapySDR::ArgInfo::FLOAT: ss << ", type=float"; break;
95  case SoapySDR::ArgInfo::STRING: ss << ", type=string"; break;
96  }
97 
98  //optional range/enumeration
99  if (argInfo.range.minimum() < argInfo.range.maximum()) ss << ", range=" << toString(argInfo.range);
100  if (not argInfo.options.empty()) ss << ", options=(" << toString(argInfo.options) << ")";
101 
102  ss << "]";
103 
104  return ss.str();
105 }
