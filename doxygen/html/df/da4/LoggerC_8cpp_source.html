74  case SOAPY_SDR_WARNING: fprintf(stderr, ANSI_COLOR_BOLD ANSI_COLOR_YELLOW "[WARNING] %s" ANSI_COLOR_RESET "\n", message); break;
75  case SOAPY_SDR_NOTICE: fprintf(stderr, ANSI_COLOR_GREEN "[NOTICE] %s" ANSI_COLOR_RESET "\n", message); break;
76  case SOAPY_SDR_INFO: fprintf(stderr, "[INFO] %s\n", message); break;
