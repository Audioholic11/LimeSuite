250 static inline void fastForward(
251  char *buff, size_t &numWritten, const size_t elemSize,
252  const uint64_t oldHeadTime, const uint64_t desiredHeadTime)
253 {
254  const size_t numPop = std::min<size_t>(desiredHeadTime - oldHeadTime, numWritten);
255  const size_t numMove = (numWritten-numPop);
256  numWritten -= numPop;
257  std::memmove(buff, buff+(numPop*elemSize), numMove*elemSize);
258 }

260 int SoapyLMS7::_readStreamAligned(
261  IConnectionStream *stream,
262  char * const *buffs,
263  size_t numElems,
264  uint64_t requestTime,
265  StreamChannel::Metadata &md,
266  const long timeoutMs)
267 {
268  const auto &streamID = stream->streamID;
269  const size_t elemSize = stream->elemSize;
270  std::vector<size_t> numWritten(streamID.size(), 0);
271 
272  for (size_t i = 0; i < streamID.size(); i += (numWritten[i]==numElems)?1:0)
273  {
274  size_t &N = numWritten[i];
275  const uint64_t expectedTime(requestTime + N);
276 
277  int status = streamID[i]->Read(buffs[i]+(elemSize*N), numElems-N,&md, timeoutMs);
278  if (status == 0) return SOAPY_SDR_TIMEOUT;
279  if (status < 0) return SOAPY_SDR_STREAM_ERROR;
280 
281  //update accounting
282  const size_t elemsRead = size_t(status);
283  const size_t prevN = N;
284  N += elemsRead; //num written total
285 
286  //unspecified request time, set the new head condition
287  if (requestTime == 0) goto updateHead;
288 
289  //good contiguous read, read again for remainder
290  if (expectedTime == md.timestamp) continue;
291 
292  //request time is later, fast forward buffer
293  if (md.timestamp < expectedTime)
294  {
295  if (prevN != 0)
296  {
297  SoapySDR::log(SOAPY_SDR_ERROR, "readStream() experienced non-monotonic timestamp");
298  return SOAPY_SDR_CORRUPTION;
299  }
300  fastForward(buffs[i], N, elemSize, md.timestamp, requestTime);
301  if (i == 0 and N != 0) numElems = N; //match size on other channels
302  continue; //read again into the remaining buffer
303  }
304 
305  //overflow in the middle of a contiguous buffer
306  //fast-forward all prior channels and restart loop
307  if (md.timestamp > expectedTime)
308  {
309  for (size_t j = 0; j < i; j++)
310  {
311  fastForward(buffs[j], numWritten[j], elemSize, requestTime, md.timestamp);
312  }
313  fastForward(buffs[i], N, elemSize, md.timestamp-prevN, md.timestamp);
314  i = 0; //start over at ch0
315  }
316 
317  updateHead:
318  requestTime = md.timestamp;
319  numElems = elemsRead;
320  }
321 
322  md.timestamp = requestTime;
323  return int(numElems);
324 }

329 int SoapyLMS7::readStream(
330  SoapySDR::Stream *stream,
331  void * const *buffs,
332  size_t numElems,
333  int &flags,
334  long long &timeNs,
335  const long timeoutUs)
336 {
337  auto icstream = (IConnectionStream *)stream;
338 
339  const auto exitTime = std::chrono::high_resolution_clock::now() + std::chrono::microseconds(timeoutUs);
340 
341  //wait for a command from activate stream up to the timeout specified
342  if (not icstream->hasCmd)
343  {
344  while (std::chrono::high_resolution_clock::now() < exitTime)
345  {
346  std::this_thread::sleep_for(std::chrono::milliseconds(10));
347  }
348  return SOAPY_SDR_TIMEOUT;
349  }
350 
351  //handle the one packet flag by clipping
352  if ((flags & SOAPY_SDR_ONE_PACKET) != 0)
353  {
354  numElems = std::min(numElems, icstream->elemMTU);
355  }
356 
357  StreamChannel::Metadata metadata;
358  const uint64_t cmdTicks = ((icstream->flags & SOAPY_SDR_HAS_TIME) != 0)?SoapySDR::timeNsToTicks(icstream->timeNs, sampleRate):0;
359  int status = _readStreamAligned(icstream, (char * const *)buffs, numElems, cmdTicks, metadata, timeoutUs/1000);
360  if (status < 0) return status;
361 
362  //the command had a time, so we need to compare it to received time
363  if ((icstream->flags & SOAPY_SDR_HAS_TIME) != 0 and (metadata.flags & RingFIFO::SYNC_TIMESTAMP) != 0)
364  {
365  //our request time is now late, clear command and return error code
366  if (cmdTicks < metadata.timestamp)
367  {
368  icstream->hasCmd = false;
369  return SOAPY_SDR_TIME_ERROR;
370  }
371 
372  //_readStreamAligned should guarantee this condition
373  if (cmdTicks != metadata.timestamp)
374  {
375  SoapySDR::logf(SOAPY_SDR_ERROR,
376  "readStream() alignment algorithm failed\n"
377  "Request time = %lld, actual time = %lld",
378  (long long)cmdTicks, (long long)metadata.timestamp);
379  return SOAPY_SDR_STREAM_ERROR;
380  }
381 
382  icstream->flags &= ~SOAPY_SDR_HAS_TIME; //clear for next read
383  }
384 
385  //handle finite burst request commands
386  if (icstream->numElems != 0)
387  {
388  //Clip to within the request size when over,
389  //and reduce the number of elements requested.
390  status = std::min<size_t>(status, icstream->numElems);
391  icstream->numElems -= status;
392 
393  //the burst completed, done with the command
394  if (icstream->numElems == 0)
395  {
396  icstream->hasCmd = false;
397  metadata.flags |= RingFIFO::END_BURST;
398  }
399  }
400 
401  //output metadata
402  flags = 0;
403  if ((metadata.flags & RingFIFO::END_BURST) != 0) flags |= SOAPY_SDR_END_BURST;
404  if ((metadata.flags & RingFIFO::SYNC_TIMESTAMP) != 0) flags |= SOAPY_SDR_HAS_TIME;
405  timeNs = SoapySDR::ticksToTimeNs(metadata.timestamp, sampleRate);
406 
407  //return num read or error code
408  return (status >= 0) ? status : SOAPY_SDR_STREAM_ERROR;
409 }
