40 std::string SoapySDR::getRootPath(void)
41 {
42  const std::string rootPathEnv = getEnvImpl("@SOAPY_SDR_ROOT_ENV@");
43  if (not rootPathEnv.empty()) return rootPathEnv;
44 
45  // Get the path to the current dynamic linked library.
46  // The path to this library can be used to determine
47  // the installation root without prior knowledge.
48  #ifdef _MSC_VER
49  char path[MAX_PATH];
50  HMODULE hm = NULL;
51  if (GetModuleHandleExA(
52  GET_MODULE_HANDLE_EX_FLAG_FROM_ADDRESS |
53  GET_MODULE_HANDLE_EX_FLAG_UNCHANGED_REFCOUNT,
54  (LPCSTR) &SoapySDR::getRootPath, &hm))
55  {
56  const DWORD size = GetModuleFileNameA(hm, path, sizeof(path));
57  if (size != 0)
58  {
59  const std::string libPath(path, size);
60  const size_t slash0Pos = libPath.find_last_of("/\\");
61  const size_t slash1Pos = libPath.substr(0, slash0Pos).find_last_of("/\\");
62  if (slash0Pos != std::string::npos and slash1Pos != std::string::npos)
63  return libPath.substr(0, slash1Pos);
64  }
65  }
66  #endif
67 
68  return "@SOAPY_SDR_ROOT@";

74 static std::vector<std::string> searchModulePath(const std::string &path)
75 {
76  std::vector<std::string> modulePaths;
77 
78 #ifdef _MSC_VER
79  const std::string pattern = path + "*.dll";
80 
81  //http://stackoverflow.com/questions/612097/how-can-i-get-a-list-of-files-in-a-directory-using-c-or-c
82  WIN32_FIND_DATA fd;
83  HANDLE hFind = ::FindFirstFile(pattern.c_str(), &fd);
84  if(hFind != INVALID_HANDLE_VALUE)
85  {
86  do
87  {
88  // read all (real) files in current folder
89  // , delete '!' read other 2 default folder . and ..
90  if(! (fd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) )
91  {
92  modulePaths.push_back(path + fd.cFileName);
93  }
94  }while(::FindNextFile(hFind, &fd));
95  ::FindClose(hFind);
96  }
97 
98 #else
99  const std::string pattern = path + "*.so";
100 
101  glob_t globResults;
102 
103  const int ret = glob(pattern.c_str(), 0/*no flags*/, NULL, &globResults);
104  if (ret == 0) for (size_t i = 0; i < globResults.gl_pathc; i++)
105  {
106  modulePaths.push_back(globResults.gl_pathv[i]);
107  }
108  else if (ret == GLOB_NOMATCH) {/* acceptable error condition, do not print error */}
109  else SoapySDR::logf(SOAPY_SDR_ERROR, "SoapySDR::listModules(%s) glob(%s) error %d", path.c_str(), pattern.c_str(), ret);
110 
111  globfree(&globResults);
112 
113 #endif
114 
115  return modulePaths;
116 }

118 std::vector<std::string> SoapySDR::listSearchPaths(void)
119 {
120  //the default search path
121  std::vector<std::string> searchPaths;
122  searchPaths.push_back(SoapySDR::getRootPath() + "/lib@LIB_SUFFIX@/SoapySDR/modules" + SoapySDR::getABIVersion());
123 
124  //support /usr/local module installs when the install prefix is /usr
125  if (SoapySDR::getRootPath() == "/usr")
126  {
127  searchPaths.push_back("/usr/local/lib@LIB_SUFFIX@/SoapySDR/modules" + SoapySDR::getABIVersion());
128  //when using a multi-arch directory, support single-arch path as well
129  static const std::string libsuffix("@LIB_SUFFIX@");
130  if (not libsuffix.empty() and libsuffix.at(0) == '/')
131  searchPaths.push_back("/usr/local/lib/SoapySDR/modules" + SoapySDR::getABIVersion());
132  }
133 
134  //separator for search paths
135  #ifdef _MSC_VER
136  static const char sep = ';';
137  #else
138  static const char sep = ':';
139  #endif
140 
141  //check the environment's search path
142  std::stringstream pluginPaths(getEnvImpl("SOAPY_SDR_PLUGIN_PATH"));
143  std::string pluginPath;
144  while (std::getline(pluginPaths, pluginPath, sep))
145  {
146  if (pluginPath.empty()) continue;
147  searchPaths.push_back(pluginPath);
148  }

226 std::string SoapySDR::loadModule(const std::string &path)
227 {
228  //check if already loaded
229  if (getModuleHandles().count(path) != 0) return path + " already loaded";
230 
231  //stash the path for registry access
232  getModuleLoading().assign(path);
233 
234  //load the module
235 #ifdef _MSC_VER
236 
237  //SetThreadErrorMode() - disable error pop-ups when DLLs are not found
238  DWORD oldMode;
239  SetThreadErrorMode(SEM_FAILCRITICALERRORS | SEM_NOGPFAULTERRORBOX | SEM_NOOPENFILEERRORBOX, &oldMode);
240  HMODULE handle = LoadLibrary(path.c_str());
241  SetThreadErrorMode(oldMode, nullptr);
242 
243  getModuleLoading().clear();
244  if (handle == NULL) return "LoadLibrary() failed: " + GetLastErrorMessage();
245 #else
246  void *handle = dlopen(path.c_str(), RTLD_LAZY);
247  getModuleLoading().clear();
248  if (handle == NULL) return "dlopen() failed: " + std::string(dlerror());
249 #endif
250 
251  //stash the handle
252  getModuleHandles()[path] = handle;
253  return "";
254 }

268 std::string SoapySDR::unloadModule(const std::string &path)
269 {
270  //check if already loaded
271  if (getModuleHandles().count(path) == 0) return path + " never loaded";
272 
273  //stash the path for registry access
274  getModuleLoading().assign(path);
275 
276  //unload the module
277  void *handle = getModuleHandles()[path];
278 #ifdef _MSC_VER
279  BOOL success = FreeLibrary((HMODULE)handle);
280  getModuleLoading().clear();
281  if (not success) return "FreeLibrary() failed: " + GetLastErrorMessage();
282 #else
283  int status = dlclose(handle);
284  getModuleLoading().clear();
285  if (status != 0) return "dlclose() failed: " + std::string(dlerror());
286 #endif
287 
288  //clear the handle
289  getLoaderResults().erase(path);
290  getModuleVersions().erase(path);
291  getModuleHandles().erase(path);
292  return "";
293 }

302 {

308  const auto paths = listModules();

315  {
316  if (it.second.empty()) continue;
317  SoapySDR::logf(SOAPY_SDR_ERROR, "SoapySDR::loadModule(%s)\n %s", paths[i].c_str(), it.second.c_str());
