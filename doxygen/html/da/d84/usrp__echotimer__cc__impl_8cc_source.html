199  usrp_echotimer_cc_impl::send()
200  {
201  // Setup metadata for first package
202  d_metadata_tx.start_of_burst = true;
203  d_metadata_tx.end_of_burst = false;
204  d_metadata_tx.has_time_spec = true;
205  d_metadata_tx.time_spec = d_time_now_tx+uhd::time_spec_t(d_wait_tx); // Timespec needed?
206 
207  // Send input buffer
208  size_t num_acc_samps = 0; // Number of accumulated samples
209  size_t num_tx_samps, total_num_samps;
210  total_num_samps = d_noutput_items_send;
211  // Data to USRP
212  num_tx_samps = d_tx_stream->send(d_in_send, total_num_samps, d_metadata_tx, total_num_samps/(float)d_samp_rate+d_timeout_tx);
213  // Get timeout
214  if (num_tx_samps < total_num_samps) std::cerr << "Send timeout..." << std::endl;
215 
216  //send a mini EOB packet
217  d_metadata_tx.start_of_burst = false;
218  d_metadata_tx.end_of_burst = true;
219  d_metadata_tx.has_time_spec = false;
220  d_tx_stream->send("", 0, d_metadata_tx);
221  }

223  void
224  usrp_echotimer_cc_impl::receive()
225  {
226  // Setup RX streaming
227  size_t total_num_samps = d_noutput_items_recv;
228  uhd::stream_cmd_t stream_cmd(uhd::stream_cmd_t::STREAM_MODE_NUM_SAMPS_AND_DONE);
229  stream_cmd.num_samps = total_num_samps;
230  stream_cmd.stream_now = false;
231  stream_cmd.time_spec = d_time_now_rx+uhd::time_spec_t(d_wait_rx);
232  d_rx_stream->issue_stream_cmd(stream_cmd);
233 
234  size_t num_rx_samps;
235  // Receive a packet
236  num_rx_samps = d_rx_stream->recv(d_out_recv, total_num_samps, d_metadata_rx, total_num_samps/(float)d_samp_rate+d_timeout_rx);
237 
238  // Save timestamp
239  d_time_val = pmt::make_tuple
240  (pmt::from_uint64(d_metadata_rx.time_spec.get_full_secs()),
241  pmt::from_double(d_metadata_rx.time_spec.get_frac_secs()));
242 
243  // Handle the error code
244  if (d_metadata_rx.error_code != uhd::rx_metadata_t::ERROR_CODE_NONE){
245  throw std::runtime_error(str(boost::format("Receiver error %s") % d_metadata_rx.strerror()));
246  }
247 
248  if (num_rx_samps < total_num_samps) std::cerr << "Receive timeout before all samples received..." << std::endl;
249  }

251  int
252  usrp_echotimer_cc_impl::work (int noutput_items,
253  gr_vector_int &ninput_items,
254  gr_vector_const_void_star &input_items,
255  gr_vector_void_star &output_items)
256  {
257  gr_complex *in = (gr_complex *) input_items[0]; // remove const
258  gr_complex *out = (gr_complex *) output_items[0];
259 
260  // Set output items on packet length
261  noutput_items = ninput_items[0];
262 
263  // Resize output buffer
264  if(d_out_buffer.size()!=noutput_items) d_out_buffer.resize(noutput_items);
265 
266  // Get time from USRP TX
267  d_time_now_tx = d_usrp_tx->get_time_now();
268  d_time_now_rx = d_time_now_tx;
269 
270  // Send thread
271  d_in_send = in;
272  d_noutput_items_send = noutput_items;
273  d_thread_send = gr::thread::thread(boost::bind(&usrp_echotimer_cc_impl::send, this));
274 
275  // Receive thread
276  d_out_recv = &d_out_buffer[0];
277  d_noutput_items_recv = noutput_items;
278  d_thread_recv = gr::thread::thread(boost::bind(&usrp_echotimer_cc_impl::receive, this));
279 
280  // Wait for threads to complete
281  d_thread_send.join();
282  d_thread_recv.join();
283 
284  // Shift of number delay samples (fill with zeros)
285  memcpy(out,&d_out_buffer[0]+d_num_delay_samps,(noutput_items-d_num_delay_samps)*sizeof(gr_complex)); // push buffer to output
286  memset(out+(noutput_items-d_num_delay_samps),0,d_num_delay_samps*sizeof(gr_complex)); // set zeros
287 
288  // Setup rx_time tag
289  add_item_tag(0, nitems_written(0), d_time_key, d_time_val, d_srcid);
290 
291  // Tell runtime system how many output items we produced.
292  return noutput_items;
293  }
