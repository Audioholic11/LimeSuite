474 void ADF4002_wxgui::OnbtnCalcSendClick(wxCommandEvent& event)
475 {
476  auto conn = ((LMS7_Device*)lmsControl)->GetConnection();
477  if (conn == nullptr)
478  {
479  wxMessageBox(_("Device not connected"), _("Error"));
480  return;
481  }
482  //reference counter latch
483  int ldp = cmbLDP->GetSelection();
484  int abw = cmbABW->GetSelection();
485  int rCount = spinRCnt->GetValue();
486  m_pModule->SetReferenceCounterLatch(ldp, abw, rCount);
487 
488  //n counter latch
489  int cpGain = cmbCPG->GetSelection();
490  int nCounter = spinNCnt->GetValue();
491  m_pModule->SetNCounterLatch(cpGain, nCounter);
492 
493  //function latch
494  int currSetting1 = cmbCS1_f->GetSelection();
495  int currSetting2 = cmbCS2_f->GetSelection();
496  int tCounter = cmbTC_f->GetSelection();
497  int fastLock = cmbFL_f->GetSelection();
498  int muxControl = cmbMOC_f->GetSelection();
499 
500  int pdPol = rgrPDP_f->GetSelection();
501  int pd1 = rgrPD1_f->GetSelection();
502  int pd2 = rgrPD2_f->GetSelection();
503  int counterReset = rgrCR_f->GetSelection();
504  int cpState = rgrCPS_f->GetSelection();
505  m_pModule->SetFunctionLatch(currSetting1, currSetting2, tCounter, fastLock, muxControl);
506  m_pModule->SetFunctionLatchRgr(pdPol, pd1,pd2, counterReset, cpState);
507 
508  //Initialization latch
509  currSetting1 = cmbCS1_i->GetSelection();
510  currSetting2 = cmbCS2_i->GetSelection();
511  tCounter = cmbTC_i->GetSelection();
512  fastLock = cmbFL_i->GetSelection();
513  muxControl = cmbMOC_i->GetSelection();
514 
515  pdPol = rgrPDP_i->GetSelection();
516  pd1 = rgrPD1_i->GetSelection();
517  pd2 = rgrPD2_i->GetSelection();
518  counterReset = rgrCR_i->GetSelection();
519  cpState = rgrCPS_i->GetSelection();
520  m_pModule->SetInitializationLatch(currSetting1, currSetting2, tCounter, fastLock, muxControl);
521  m_pModule->SetInitializationLatchRgr(pdPol, pd1,pd2, counterReset, cpState);
522 
523  double fref = 0;
524  txtFref->GetValue().ToDouble(&fref);
525 
526  double fvco = 0;
527  txtFvco->GetValue().ToDouble(&fvco);
528  m_pModule->SetFrefFvco(fref, fvco, rCount, nCounter);
529  spinNCnt->SetValue(nCounter);
530  spinRCnt->SetValue(rCount);
531 
532  lblFcomp->SetLabel( wxString::Format("%f", m_pModule->lblFcomp));
533  lblFvco->SetLabel( wxString::Format("%f", m_pModule->lblFvco));
534 
535  unsigned char data[12];
536  m_pModule->GetConfig(data);
537 
538  vector<uint32_t> dataWr;
539  for(int i=0; i<12; i+=3)
540  dataWr.push_back((uint32_t)data[i] << 16 | (uint32_t)data[i+1] << 8 | data[i+2]);
541 
542  int status;
543 // ADF4002 needs to be writen 4 values of 24 bits
544  status = conn->TransactSPI(0x30, dataWr.data(), nullptr, 4);
545  if (status != 0)
546  wxMessageBox(_("ADF configuration failed"), _("Error"));
547 }
