69  split_fsk_cc_impl::work (int noutput_items,
70  gr_vector_int &ninput_items,
71  gr_vector_const_void_star &input_items,
72  gr_vector_void_star &output_items)
73  {
74  const gr_complex *in = (const gr_complex *) input_items[0];
75  gr_complex *out0 = (gr_complex *) output_items[0];
76  gr_complex *out1 = (gr_complex *) output_items[1];
77 
78  // get all tags, reset offset and push to outputs
79  get_tags_in_range(d_tags,0,nitems_read(0),nitems_read(0)+1);
80  for(int k=0; k<d_tags.size(); k++){
81  add_item_tag(0,nitems_written(0),d_tags[k].key,d_tags[k].value,d_tags[k].srcid);
82  add_item_tag(1,nitems_written(1),d_tags[k].key,d_tags[k].value,d_tags[k].srcid);
83  }
84 
85  // Resize packet len and set noutput_items
86  d_nblocks = ninput_items[0]/2/d_samp_per_freq;
87  noutput_items = d_nblocks*(d_samp_per_freq-d_samp_discard);
88  update_length_tags(noutput_items,2); // update length tag
89 
90  // Push input_items to outputs
91  int k=0;
92  while(k<ninput_items[0]){
93  // first freq
94  k+=d_samp_discard;
95  for(int l=0; l<d_samp_per_freq-d_samp_discard; l++) *out0++ = in[k+l];
96  k+=d_samp_per_freq-d_samp_discard;
97  // second freq
98  k+=d_samp_discard;
99  for(int l=0; l<d_samp_per_freq-d_samp_discard; l++) *out1++ = in[k+l];
100  k+=d_samp_per_freq-d_samp_discard;
101  }
102 
103  // Tell runtime system how many output items we produced.
104  return noutput_items;
105  }
