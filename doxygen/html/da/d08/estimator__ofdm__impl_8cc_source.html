72  estimator_ofdm_impl::handle_msg(pmt::pmt_t msg)
73  {
74  // Read msg from peak detector
75  pmt::pmt_t msg_part;
76  for(int k=0; k<pmt::length(msg); k++){
77  msg_part = pmt::nth(k,msg);
78  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="axis_x"){
79  d_paxis_x = pmt::nth(1,msg_part);
80  }
81  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="axis_y"){
82  d_paxis_y = pmt::nth(1,msg_part);
83  }
84  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="power"){
85  d_ppower = pmt::nth(1,msg_part);
86  }
87  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="rx_time"){
88  d_ptimestamp = msg_part;
89  }
90  }
91 
92  d_val_x = pmt::f32vector_elements(d_paxis_x);
93  d_val_y = pmt::f32vector_elements(d_paxis_y);
94 
95  // Merge consecutive peaks
96  std::vector<float> val_x_hold, val_y_hold;
97  val_x_hold = d_val_x;
98  val_y_hold = d_val_y;
99  if(d_merge_consecutive){
100  d_val_power = pmt::f32vector_elements(d_ppower);
101  for(int k=0; k<d_val_x.size(); k++){ // Go through points and check bins in range one (also diagonal!) if the power is greater then the CUT
102  for(int l=0; l<d_val_x.size(); l++){
103  if(std::abs(d_val_x[k]-d_val_x[l])<=1 && std::abs(d_val_y[k]-d_val_y[l])<=1 && d_val_power[k]<d_val_power[l]){
104  val_x_hold[k] = -1;
105  val_y_hold[k] = -1;
106  }
107  }
108  }
109  d_val_x.clear();
110  d_val_y.clear();
111  for(int k=0; k<val_x_hold.size(); k++){
112  if(val_x_hold[k]!=-1 && val_y_hold[k]!=-1){
113  d_val_x.push_back(val_x_hold[k]);
114  d_val_y.push_back(val_y_hold[k]);
115  }
116  }
117  }
118 
119  // Map bins on given axis
120  // 4 values given splits in the middle of the axis and interpolates linear in between
121  // 2 values given represents min and max with linear progression in between
122  d_map_x.clear();
123  d_map_y.clear();
124  for(int k=0; k<d_val_x.size(); k++){ // x axis
125  if(d_axis_x.size()==2){
126  d_map_x.push_back(d_axis_x[0]+d_val_x[k]/d_len_x*(d_axis_x[1]-d_axis_x[0]));
127  }
128  else if(d_axis_x.size()==4){
129  if(d_val_x[k]<d_len_x/2){ // lower part
130  d_map_x.push_back(d_axis_x[0]+d_val_x[k]/(float)d_len_x*2.0*(d_axis_x[1]-d_axis_x[0]));
131  }
132  else{ // upper part
133  d_map_x.push_back(d_axis_x[2]+(d_val_x[k]-d_len_x/2.0)/(float)d_len_x*2.0*(d_axis_x[3]-d_axis_x[2]));
134  }
135  }
136  }
137 
138  for(int k=0; k<d_val_y.size(); k++){ // y axis
139  if(d_axis_y.size()==2){
140  d_map_y.push_back(d_axis_y[0]+d_val_y[k]/d_len_y*(d_axis_y[1]-d_axis_y[0]));
141  }
142  else if(d_axis_y.size()==4){
143  if(d_val_y[k]<d_len_y/2){ // lower part
144  d_map_y.push_back(d_axis_y[0]+d_val_y[k]/(float)d_len_y*2.0*(d_axis_y[1]-d_axis_y[0]));
145  }
146  else{ // upper part
147  d_map_y.push_back(d_axis_y[2]+(d_val_y[k]-d_len_y/2.0)/(float)d_len_y*2.0*(d_axis_y[3]-d_axis_y[2]));
148  }
149  }
150  }
151 
152  // Push pmt to output msg port
153  d_x_key = pmt::string_to_symbol(d_symbol_x); // identifier x axis
154  d_x_value = pmt::init_f32vector(d_map_x.size(), d_map_x); // vector to pmt
155  d_x_pack = pmt::list2(d_x_key, d_x_value); // make list for x axis information
156 
157  d_y_key = pmt::string_to_symbol(d_symbol_y); // identifier y axis
158  d_y_value = pmt::init_f32vector(d_map_y.size(), d_map_y); // vector to pmt
159  d_y_pack = pmt::list2(d_y_key, d_y_value); // make list for y axis information
160 
161  d_value = pmt::list3(d_ptimestamp, d_x_pack, d_y_pack);
162 
163  message_port_pub(d_port_id_out,d_value); // publish message
164  }
