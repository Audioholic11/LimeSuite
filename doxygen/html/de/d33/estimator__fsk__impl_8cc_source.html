67  void
68  estimator_fsk_impl::handle_msg(pmt::pmt_t msg)
69  {
70  // Read msg from peak detector
71  pmt::pmt_t msg_part;
72  for(int k=0; k<pmt::length(msg); k++){
73  msg_part = pmt::nth(k,msg);
74  if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="frequency"){
75  d_pfreq = pmt::nth(1,msg_part);
76  }
77  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="phase"){
78  d_pphase = pmt::nth(1,msg_part);
79  }
80  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="power"){
81  d_ppower = msg_part;
82  }
83  else if(pmt::symbol_to_string(pmt::nth(0,msg_part))=="rx_time"){
84  d_ptimestamp = msg_part;
85  }
86  }
87 
88  d_freq = pmt::f32vector_elements(d_pfreq);
89  d_phase = pmt::f32vector_elements(d_pphase);
90 
91  // Check read data
92  if(d_phase.size()==0 && d_freq.size()!=0) std::runtime_error("Frequency but no phase found in message");
93 
94  // Calc velocities and write to vector
95  d_vel.clear();
96  for(int k=0; k<d_freq.size(); k++){
97  d_vel.push_back(d_freq[k]*c_light/2/d_center_freq); // calc with doppler formula
98  }
99 
100  // Calc ranges and write to vector
101  d_range.clear();
102  for(int k=0; k<d_phase.size(); k++){
103  if(d_phase[k]>=0) d_range.push_back((d_phase[k])*c_light/4/M_PI/d_delta_freq); // calc with fsk range formula
104  else if(d_phase[k]<0) d_range.push_back((2*M_PI+d_phase[k])*c_light/4/M_PI/d_delta_freq); // phase jumps from pi to -pi
105  }
106 
107  // Push pmt to output msg port
108  d_vel_key = pmt::string_to_symbol("velocity"); // identifier velocity
109  d_vel_value = pmt::init_f32vector(d_vel.size(), d_vel); // vector to pmt
110  d_vel_pack = pmt::list2(d_vel_key, d_vel_value); // make list for velocity information
111 
112  d_range_key = pmt::string_to_symbol("range"); // identifier range
113  d_range_value = pmt::init_f32vector(d_range.size(), d_range); // vector to pmt
114  d_range_pack = pmt::list2(d_range_key, d_range_value); // make list for range information
115 
116  if(d_push_power){ // if power of peaks shall be pushed through
117  d_value = pmt::list4(d_ptimestamp, d_vel_pack, d_range_pack, d_ppower); // all information to one pmt list
118  }
119  else{
120  d_value = pmt::list3(d_ptimestamp, d_vel_pack, d_range_pack);
121  }
122 
123  message_port_pub(d_port_id_out,d_value); // publish message
124  }
