79  os_cfar_c_impl::work (int noutput_items,
80  gr_vector_int &ninput_items,
81  gr_vector_const_void_star &input_items,
82  gr_vector_void_star &output_items)
83  {
84  const gr_complex *in = (const gr_complex *) input_items[0];
85 
86  // OS-CFAR detection
87  d_freq.clear();
88  d_pks.clear();
89  d_angle.clear();
90 
91  for(int k=0; k<ninput_items[0]; k++){ // go through input
92  d_hold_samp.clear();
93  for(int l=1; l<d_samp_compare+1; l++){ // go through samples to compare, care: num of samp_compare is doubled! redef if too confusing
94  if(k-l-d_samp_protect<0){ // push-back zeros for underflows
95  d_hold_samp.push_back(0);
96  }
97  else{ // push back abs-square
98  d_hold_samp.push_back(std::pow(std::abs(in[k-l-d_samp_protect]),2));
99  }
100  if(k+l+d_samp_protect>=ninput_items[0]){ // push-back zeros for overflows
101  d_hold_samp.push_back(0);
102  }
103  else{ // push back abs-square
104  d_hold_samp.push_back(std::pow(std::abs(in[k+l+d_samp_protect]),2));
105  }
106  }
107  std::sort(d_hold_samp.begin(),d_hold_samp.end()); // sort sample vector
108 
109  if(std::pow(std::abs(in[k]),2)>d_hold_samp[(int)((2*d_samp_compare-1)*d_rel_threshold)]*d_mult_threshold){ // check if in[k] is over dynamic threshold multiplied with mult_threshold
110  // Add peaks, frequencies and angle
111  if(k<=ninput_items[0]/2){
112  d_freq.push_back(k*(d_samp_rate/(float)ninput_items[0])); // add frequency to message vector d_freq
113  }
114  else{
115  d_freq.push_back(-(float)d_samp_rate+k*(d_samp_rate/(float)ninput_items[0]));
116  }
117  d_pks.push_back(pow(abs(in[k]),2)); // add abs-square to message vector d_pks
118  d_angle.push_back(std::arg(in[k])); // add angle to message vector d_angle
119 
120  // Merge consecutive peaks
121  if( (d_merge_consecutive && d_consecutive) && d_freq.size()>1){
122  if(d_pks[d_pks.size()-2]<d_pks[d_pks.size()-1]){ // if last peak is lower, remove last one
123  d_pks.erase(d_pks.end()-2); // vec.end() points past the end!
124  d_freq.erase(d_freq.end()-2);
125  d_angle.erase(d_angle.end()-2);
126  }
127  else{ // if last peak is higher, remove actual one
128  d_pks.erase(d_pks.end()-1);
129  d_freq.erase(d_freq.end()-1);
130  d_angle.erase(d_angle.end()-1);
131  }
132  }
133 
134  d_consecutive = true; // tag next loop cycle with check for merging peaks
135  }
136  else{
137  d_consecutive = false; // release consecutive peaks merger if no peak is detected
138  }
139 
140  }
141 
142  // get rx_time tag
143  get_tags_in_range(d_tags,0,nitems_read(0),nitems_read(0)+1,pmt::string_to_symbol("rx_time"));
144 
145  // setup msg pmt
146  if(d_tags.size()>0) d_ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),d_tags[0].value);
147  else d_ptimestamp = pmt::list2(pmt::string_to_symbol("rx_time"),pmt::make_tuple(pmt::from_uint64(0),pmt::from_double(-1))); // if no timetag is found, set to 0 and frac_sec to -1
148  d_pfreq = pmt::list2(pmt::string_to_symbol("frequency"),pmt::init_f32vector(d_freq.size(),d_freq));
149  d_ppks = pmt::list2(pmt::string_to_symbol("power"),pmt::init_f32vector(d_pks.size(),d_pks));
150  d_pangle = pmt::list2(pmt::string_to_symbol("phase"),pmt::init_f32vector(d_angle.size(),d_angle));
151  d_value = pmt::list4(d_ptimestamp,d_pfreq,d_ppks,d_pangle);
152 
153  // publish message
154  message_port_pub(d_port_id,d_value);
155 
156  // Tell runtime system how many output items we produced.
157  return 0;
158  }
