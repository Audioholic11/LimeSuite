592  std::this_thread::sleep_for(std::chrono::milliseconds(500));

596  else

662  if(rxRate_Hz >= 5e6 && txRate_Hz >= 5e6)

664  uint32_t addr[3] = {0, 1, 2};
665  uint32_t vals[3];

667  vals[1] = (vals[1]<<8)|vals[2];

669  phaseSearch = true;
670  }
671 
672  if (!phaseSearch)
673  return SetInterfaceFreq(txRate_Hz, rxRate_Hz, txPhC1 + txPhC2 * txRate_Hz, rxPhC1 + rxPhC2 * rxRate_Hz, channel);
674 
675  std::vector<uint32_t> dataRdA;
676  std::vector<uint32_t> dataRdB;
677  std::vector<uint32_t> dataWr;
678 
679  dataWr.resize(spiAddr.size());
680  dataRdA.resize(bakRegCnt);
681  dataRdB.clear();
682  //backup registers
683  dataWr[0] = (uint32_t(0x0020) << 16);
684  connection->ReadLMS7002MSPI(dataWr.data(), &reg20, 1, channel);
685 
686  dataWr[0] = (1 << 31) | (uint32_t(0x0020) << 16) | 0xFFFD; //msbit 1=SPI write
687  connection->WriteLMS7002MSPI(dataWr.data(), 1, channel);
688 
689  for (int i = 0; i < bakRegCnt; ++i)
690  dataWr[i] = (spiAddr[i] << 16);
691  connection->ReadLMS7002MSPI(dataWr.data(),dataRdA.data(), bakRegCnt, channel);
692 
693  {
694  const uint32_t addr = (0x02A<<16);
695  uint32_t val;
696  connection->ReadLMS7002MSPI(&addr, &val, 1, channel);
697  bypassTx = (val&0xF0) == 0x00;
698  bypassRx = (val&0x0F) == 0x0D;
699  }
700 
701  dataWr[0] = (1 << 31) | (uint32_t(0x0020) << 16) | 0xFFFE; //msbit 1=SPI write
702  connection->WriteLMS7002MSPI(dataWr.data(), 1, channel);
703 
704  for (int i = 0; i < bakRegCnt; ++i)
705  if (spiAddr[i] >= 0x100)
706  dataRdB.push_back(spiAddr[i] << 16);
707  connection->ReadLMS7002MSPI(dataRdB.data(), dataRdB.data(), dataRdB.size(), channel);
708 
709  dataWr[0] = (1 << 31) | (uint32_t(0x0020) << 16) | 0xFFFF; //msbit 1=SPI write
710  connection->WriteLMS7002MSPI(dataWr.data(), 1, channel);
711 
712  {
713  std::vector<uint32_t> spiData = { 0x0E9F, 0x0FFF, 0x5550, 0xE4E4, 0xE4E4, 0x0086, 0x8001,
714  0x028D, 0x00FF, 0x5555, 0x02CD, 0xAAAA, 0x02ED};
715  if (bypassRx)spiData[5] = 0xD;
716  //Load test config
717  const int setRegCnt = spiData.size();
718  for (int i = 0; i < setRegCnt; ++i)
719  dataWr[i] = (1 << 31) | (uint32_t(spiAddr[i]) << 16) | spiData[i]; //msbit 1=SPI write
720  connection->WriteLMS7002MSPI(dataWr.data(), setRegCnt, channel);
721  }
722 
723  lime::FPGA::FPGA_PLL_clock clocks[2];
724  clocks[0].index = 1;
725  clocks[0].outFrequency = bypassRx ? 2*rxRate_Hz : rxRate_Hz;
726  clocks[0].phaseShift_deg = rxPhC1 + rxPhC2 * rxRate_Hz;
727  clocks[0].findPhase = true;
728  clocks[1] = clocks[0];
729  if (SetPllFrequency(pll_ind+1, rxRate_Hz, clocks, 2)!=0)
730  {
731  status = -1;
732  clocks[0].index = 0;
733  clocks[0].phaseShift_deg = 0;
734  clocks[0].findPhase = false;
735  clocks[1].findPhase = false;
736  SetPllFrequency(pll_ind+1, rxRate_Hz, clocks, 2);
737  }
738 
739  {
740  std::vector<uint32_t> spiData = {0x0E9F, 0x0FFF, 0x5550, 0xE4E4, 0xE4E4, 0x0484, 0x8001};
741  if (bypassTx)spiData[5] ^= 0x80;
742  if (bypassRx)spiData[5] ^= 0x9;
743  connection->WriteRegister(0xFFFF, 1 << channel);
744  connection->WriteRegister(0x000A, 0x0000);
745  //Load test config
746  const int setRegCnt = spiData.size();
747  for (int i = 0; i < setRegCnt; ++i)
748  dataWr[i] = (1 << 31) | (uint32_t(spiAddr[i]) << 16) | spiData[i]; //msbit 1=SPI write
749  connection->WriteLMS7002MSPI(dataWr.data(), setRegCnt, channel);
750  }
751 
752  clocks[0].index = 1;
753  clocks[0].outFrequency = bypassTx ? 2*txRate_Hz:txRate_Hz;
754  clocks[0].phaseShift_deg = txPhC1 + txPhC2 * txRate_Hz;
755  clocks[0].findPhase = true;
756  clocks[1] = clocks[0];
757  connection->WriteRegister(0x000A, 0x0200);
758  if (SetPllFrequency(pll_ind, txRate_Hz, clocks, 2)!=0)

762  clocks[0].phaseShift_deg = 0;
763  clocks[0].findPhase = false;
764  clocks[1].findPhase = false;
765  SetPllFrequency(pll_ind, txRate_Hz, clocks, 2);
766  }
767 
768  //Restore registers
769  dataWr[0] = (1 << 31) | (uint32_t(0x0020) << 16) | 0xFFFD; //msbit 1=SPI write
770  connection->WriteLMS7002MSPI(dataWr.data(), 1, channel);
771  for (int i = 0; i < bakRegCnt; ++i)
772  dataWr[i] = (1 << 31) | (uint32_t(spiAddr[i]) << 16) | dataRdA[i]; //msbit 1=SPI write
773  connection->WriteLMS7002MSPI(dataWr.data(), bakRegCnt, channel);
774  dataWr[0] = (1 << 31) | (uint32_t(0x0020) << 16) | 0xFFFE; //msbit 1=SPI write
775  connection->WriteLMS7002MSPI(dataWr.data(), 1, channel);

788  return status;

805 double FPGA::DetectRefClk(double fx3Clk)
806 {
807  const double fx3Cnt = 16777210; //fixed fx3 counter in FPGA
808  const double clkTbl[] = { 30.72e6, 38.4e6, 40e6, 52e6 };
809  const uint32_t addr[] = { 0x61, 0x63 };
810  const uint32_t vals[] = { 0x0, 0x0 };
811  if (connection->WriteRegisters(addr, vals, 2) != 0)
812  return -1;
813 
814  auto start = std::chrono::steady_clock::now();
815  if (connection->WriteRegister(0x61, 0x4) != 0)
816  return -1;
817 
818  while (1) //wait for test to finish
819  {
820  unsigned completed;
821  if (connection->ReadRegister(0x65, completed) != 0)
822  return -1;
823 
824  if (completed & 0x4)
825  break;
826 
827  auto end = std::chrono::steady_clock::now();
828  std::chrono::duration<double> elapsed_seconds = end - start;
829  if (elapsed_seconds.count() > 0.5) //timeout
830  return -1;
831  }
832 
833  const uint32_t addr2[] = { 0x72, 0x73 };
834  uint32_t vals2[2];
835  if (connection->ReadRegisters(addr2, vals2, 2) != 0)
836  return -1;
837 
838  double count = (vals2[0] | (vals2[1] << 16)); //cock counter
839  count *= fx3Clk / fx3Cnt; //estimate ref clock based on FX3 Clock
840  lime::debug("Estimated reference clock %1.4f MHz", count/1e6);
841  unsigned i = 0;
842  double delta = 100e6;
843 
844  while (i < sizeof(clkTbl) / sizeof(*clkTbl))
845  if (delta < fabs(count - clkTbl[i]))
846  break;
847  else
848  delta = fabs(count - clkTbl[i++]);
849 
850  lime::info("Reference clock %1.2f MHz", clkTbl[i - 1] / 1e6);
851  return clkTbl[i - 1];
852 }
