840 {

843  unsigned int address, length;

845 
846  if ( filesize > MAX_FWIMG_SIZE ) {

848  return -2;

863  return -6;
864  }
865 
866  // Run through each section of code, and use vendor commands to download them to RAM.
867  index = 4;
868  checksum = 0;

872  length = data_p[0];

874  if (length != 0)
875  {

879  if (r != 0)

881  ReportError("Failed to download data to FX3 RAM\n");
882  return -3;

884  }
885  else
886  {
887  if (checksum != data_p[2]) {
888  ReportError ("Checksum error in firmware binary\n");
889  return -4;
890  }
891

893  if ( r != 0 )
894  lime::error("Ignored error in control transfer: %d", r);
895  break;
