579 static uint8_t SearchTxFilterCCAL_RCAL(uint16_t addr, uint8_t msblsb)
580 {
581  ROM const ReachRSSIparams paramLPF_LAD_decrease = {CCAL_LPFLAD_TBB, -1, 0, true};
582  ROM const ReachRSSIparams paramLPF_LAD_increase = {CCAL_LPFLAD_TBB, 1, 31, false};
583  bool targetLevelReached;
584  uint8_t iterationsLeft = 6;
585  do
586  {
587  uint16_t rssi_3dB_lad;
588  uint8_t ccal_limit;
589  int8_t rcal_step;
590  bool rssiShouldBeLess;
591  bool needToChangeCCAL;
592  Modify_SPI_Reg_bits(SEL_TX, 0);
593  Modify_SPI_Reg_bits(SEL_RX, 0);
594  rssi_3dB_lad = GetRSSI() * 0.7071;
595  Modify_SPI_Reg_bits(SEL_TX, 1);
596  Modify_SPI_Reg_bits(SEL_RX, 1);
597 
598  if(GetRSSI() < rssi_3dB_lad)
599  {
600  ccal_limit = 0;
601  rcal_step = 25;
602  rssiShouldBeLess = true;
603  needToChangeCCAL = false;
604  }
605  else
606  {
607  ccal_limit = 31;
608  rcal_step = -10;
609  rssiShouldBeLess = false;
610  needToChangeCCAL = true;
611  }
612 
613  if(rssiShouldBeLess)
614  targetLevelReached = ChangeUntilReachRSSI(&paramLPF_LAD_decrease, rssi_3dB_lad);
615  else
616  targetLevelReached = ChangeUntilReachRSSI(&paramLPF_LAD_increase, rssi_3dB_lad);
617  if( !targetLevelReached && Get_SPI_Reg_bits(CCAL_LPFLAD_TBB) == ccal_limit)
618  {
619  uint8_t R = Get_SPI_Reg_bits(addr, msblsb);
620  if(R == 0 || R == 255)
621  return MCU_NO_ERROR; // reached filter bandwidth limit
622  Modify_SPI_Reg_bits(addr, msblsb, clamp(R+rcal_step, 0, 255));
623  Modify_SPI_Reg_bits(CCAL_LPFLAD_TBB, 16);
624  }
625  else if(needToChangeCCAL)
626  {
627  uint8_t ccal_lpflad_tbb = Get_SPI_Reg_bits(CCAL_LPFLAD_TBB);
628  ccal_lpflad_tbb = clamp(++ccal_lpflad_tbb, 0, 31);
629  Modify_SPI_Reg_bits(CCAL_LPFLAD_TBB, ccal_lpflad_tbb);
630  }
631  --iterationsLeft;
632  } while(!targetLevelReached && iterationsLeft);
633  return targetLevelReached ? MCU_NO_ERROR : MCU_RCAL_LPF_LIMIT_REACHED;
634 }

636 uint8_t TuneTxFilter(const float_type tx_lpf_freq_RF)
637 {
638  float_type tx_lpf_IF;
639  uint8_t status;
640 
641  if(tx_lpf_freq_RF < TxLPF_RF_LimitLow || tx_lpf_freq_RF > TxLPF_RF_LimitHigh)
642  return MCU_TX_LPF_OUT_OF_RANGE;
643  //calculate intermediate frequency
644  tx_lpf_IF = tx_lpf_freq_RF/2;
645  if(tx_lpf_freq_RF > TxLPF_RF_LimitLowMid && tx_lpf_freq_RF < TxLPF_RF_LimitMidHigh)
646  tx_lpf_IF = TxLPF_RF_LimitMidHigh/2;
647  SaveChipState(0);

652 
653  Modify_SPI_Reg_bits(SEL_RX, 0);
654  Modify_SPI_Reg_bits(SEL_TX, 0);
655  {
656  ROM const ReachRSSIparams paramTX = {CG_IAMP_TBB, 1, 43, true};
657  ChangeUntilReachRSSI(&paramTX, 0x2700);
658  }
659 
660  //LPFL
661  if(tx_lpf_IF <= TxLPF_RF_LimitLowMid/2)
662  status = SearchTxFilterCCAL_RCAL(RCAL_LPFLAD_TBB);
663  else // LPFH
664  {
665  Modify_SPI_Reg_bits(C_CTL_PGA_RBB, 2);
666  status = SearchTxFilterCCAL_RCAL(RCAL_LPFH_TBB);
667  }
668  {
669  uint16_t ccal_lpflad_tbb = Get_SPI_Reg_bits(CCAL_LPFLAD_TBB);
670  uint16_t rcal_lpfh_lpflad_tbb = SPI_read(0x0109);
671  TxFilterSearchEndStage:
672  SaveChipState(1);
673  if(status != MCU_NO_ERROR)
674  return status;
675  Modify_SPI_Reg_bits(CCAL_LPFLAD_TBB, ccal_lpflad_tbb);
676  SPI_write(0x0106, 0x318C);
677  SPI_write(0x0107, 0x318C);
678  SPI_write(0x0109, rcal_lpfh_lpflad_tbb);
679  SetTxLPF_PDs(tx_lpf_IF);
680  }
681 
682  return MCU_NO_ERROR;
683 }
