356 int LMS64CProtocol::TransferPacket(GenericPacket& pkt)
357 {
358  std::lock_guard<std::mutex> lock(mControlPortLock);
359  int status = 0;
360  if(IsOpen() == false) ReportError(ENOTCONN, "connection is not open");
361 
362  int packetLen;
363  eLMS_PROTOCOL protocol = LMS_PROTOCOL_UNDEFINED;
364  if(this->GetType() == SPI_PORT)
365  protocol = LMS_PROTOCOL_NOVENA;
366  else
367  protocol = LMS_PROTOCOL_LMS64C;
368  switch(protocol)
369  {
370  case LMS_PROTOCOL_UNDEFINED:
371  return ReportError("protocol type undefined");
372  case LMS_PROTOCOL_LMS64C:
373  packetLen = ProtocolLMS64C::pktLength;
374  break;
375  case LMS_PROTOCOL_NOVENA:
376  packetLen = pkt.outBuffer.size() > ProtocolNovena::pktLength ? ProtocolNovena::pktLength : pkt.outBuffer.size();
377  break;
378  default:
379  packetLen = 0;
380  return ReportError("Unknown protocol type %d", int(protocol));
381  }
382  int outLen = 0;
383  unsigned char* outBuffer = NULL;
384  outBuffer = PreparePacket(pkt, outLen, protocol);
385  unsigned char* inBuffer = new unsigned char[outLen];
386  memset(inBuffer, 0, outLen);
387 
388  int outBufPos = 0;
389  int inDataPos = 0;
390  if(outLen == 0)
391  outLen = 1;
392 
393  if(protocol == LMS_PROTOCOL_NOVENA)
394  {
395  bool transferData = true; //some commands are fake, so don't need transferring
396  if(pkt.cmd == CMD_GET_INFO)
397  {
398  //spi does not have GET INFO, fake it to inform what device it is
399  pkt.status = STATUS_COMPLETED_CMD;
400  pkt.inBuffer.clear();
401  pkt.inBuffer.resize(64, 0);
402  pkt.inBuffer[0] = 0; //firmware
403  pkt.inBuffer[1] = LMS_DEV_NOVENA; //device
404  pkt.inBuffer[2] = 0; //protocol
405  pkt.inBuffer[3] = 0; //hardware
406  pkt.inBuffer[4] = EXP_BOARD_UNSUPPORTED; //expansion
407  transferData = false;
408  }
409 
410  if(transferData)
411  {
412  if (callback_logData)
413  callback_logData(true, outBuffer, outLen);
414  int bytesWritten = Write(outBuffer, outLen);
415  if( bytesWritten == outLen)
416  {
417  if(pkt.cmd == CMD_LMS7002_RD)
418  {
419  inDataPos = Read(&inBuffer[inDataPos], outLen);
420  if(inDataPos != outLen)
421  status = ReportError("Read(%d bytes) got %d", (int)outLen, (int)inDataPos);
422  else
423  {
424  if (callback_logData)
425  callback_logData(false, inBuffer, inDataPos);
426  }
427  }
428  ParsePacket(pkt, inBuffer, inDataPos, protocol);
429  }
430  else
431  status = ReportError("Write(%d bytes) got %d", (int)outLen, (int)bytesWritten);
432  }
433  }
434  else
435  {
436  for(int i=0; i<outLen; i+=packetLen)
437  {
438  int bytesToSend = packetLen;
439  if (callback_logData)
440  callback_logData(true, &outBuffer[outBufPos], bytesToSend);
441  if( Write(&outBuffer[outBufPos], bytesToSend) )
442  {
443  outBufPos += packetLen;
444  long readLen = packetLen;
445  int bread = Read(&inBuffer[inDataPos], readLen);
446  if(bread != readLen && protocol != LMS_PROTOCOL_NOVENA)
447  {
448  status = ReportError(EIO, "Read(%d bytes) failed", (int)readLen);
449  break;
450  }
451  if (callback_logData)
452  callback_logData(false, &inBuffer[inDataPos], bread);
453  inDataPos += bread;
454  }
455  else
456  {
457  status = ReportError(EIO, "Write(%d bytes) failed", (int)bytesToSend);
458  break;
459  }
460  }
461  ParsePacket(pkt, inBuffer, inDataPos, protocol);
462  }
463  delete[] outBuffer;
464  delete[] inBuffer;
465  return convertStatus(status, pkt);
466 }

474 unsigned char* LMS64CProtocol::PreparePacket(const GenericPacket& pkt, int& length, const eLMS_PROTOCOL protocol)
475 {
476  unsigned char* buffer = NULL;
477  if(protocol == LMS_PROTOCOL_UNDEFINED)
478  return NULL;
479 
480  if(protocol == LMS_PROTOCOL_LMS64C)
481  {
482  ProtocolLMS64C packet;
483  int maxDataLength = packet.maxDataLength;
484  packet.cmd = pkt.cmd;
485  packet.status = pkt.status;
486  packet.periphID = pkt.periphID;
487  int byteBlockRatio = 1; //block ratio - how many bytes in one block
488  switch( packet.cmd )
489  {
490  case CMD_PROG_MCU:
491  case CMD_GET_INFO:
492  case CMD_SI5351_RD:
493  case CMD_SI5356_RD:
494  byteBlockRatio = 1;
495  break;
496  case CMD_SI5351_WR:
497  case CMD_SI5356_WR:
498  byteBlockRatio = 2;
499  break;
500  case CMD_LMS7002_RD:
501  case CMD_BRDSPI_RD:
502  case CMD_BRDSPI8_RD:
503  byteBlockRatio = 2;
504  break;
505  case CMD_ADF4002_WR:
506  byteBlockRatio = 3;
507  break;
508  case CMD_LMS7002_WR:
509  case CMD_BRDSPI_WR:
510  case CMD_ANALOG_VAL_WR:
511  byteBlockRatio = 4;
512  break;
513  default:
514  byteBlockRatio = 1;
515  }
516  if (packet.cmd == CMD_LMS7002_RD || packet.cmd == CMD_BRDSPI_RD)
517  maxDataLength = maxDataLength/2;
518  if (packet.cmd == CMD_ANALOG_VAL_RD)
519  maxDataLength = maxDataLength / 4;
520  int blockCount = pkt.outBuffer.size()/byteBlockRatio;
521  int bufLen = blockCount/(maxDataLength/byteBlockRatio)
522  +(blockCount%(maxDataLength/byteBlockRatio)!=0);
523  bufLen *= packet.pktLength;
524  if(bufLen == 0)
525  bufLen = packet.pktLength;
526  buffer = new unsigned char[bufLen];
527  memset(buffer, 0, bufLen);
528  unsigned int srcPos = 0;
529  for(int j=0; j*packet.pktLength<bufLen; ++j)
530  {
531  int pktPos = j*packet.pktLength;
532  buffer[pktPos] = packet.cmd;
533  buffer[pktPos+1] = packet.status;
534  buffer[pktPos + 3] = packet.periphID;
535  if(blockCount > (maxDataLength/byteBlockRatio))
536  {
537  buffer[pktPos+2] = maxDataLength/byteBlockRatio;
538  blockCount -= buffer[pktPos+2];
539  }
540  else
541  buffer[pktPos+2] = blockCount;
542  memcpy(&buffer[pktPos+4], packet.reserved, sizeof(packet.reserved));
543  int bytesToPack = (maxDataLength/byteBlockRatio)*byteBlockRatio;
544  for (int k = 0; k<bytesToPack && srcPos < pkt.outBuffer.size(); ++srcPos, ++k)
545  buffer[pktPos + 8 + k] = pkt.outBuffer[srcPos];
546  }
547  length = bufLen;
548  }
549  else if(protocol == LMS_PROTOCOL_NOVENA)
550  {
551  if(pkt.cmd == CMD_LMS7002_RST)
552  {
553  buffer = new unsigned char[8];
554  buffer[0] = 0x88;
555  buffer[1] = 0x06;
556  buffer[2] = 0x00;
557  buffer[3] = 0x18;
558  buffer[4] = 0x88;
559  buffer[5] = 0x06;
560  buffer[6] = 0x00;
561  buffer[7] = 0x38;
562  length = 8;
563  }
564  else
565  {
566  buffer = new unsigned char[pkt.outBuffer.size()];
567  memcpy(buffer, &pkt.outBuffer[0], pkt.outBuffer.size());
568  if (pkt.cmd == CMD_LMS7002_WR)
569  {
570  for(size_t i=0; i<pkt.outBuffer.size(); i+=4)
571  buffer[i] |= 0x80;
572  }
573  length = pkt.outBuffer.size();
574  }
575  }
576  return buffer;
577 }

586 int LMS64CProtocol::ParsePacket(GenericPacket& pkt, const unsigned char* buffer, const int length, const eLMS_PROTOCOL protocol)
587 {
588  if(protocol == LMS_PROTOCOL_UNDEFINED)
589  return -1;
590 
591  if(protocol == LMS_PROTOCOL_LMS64C)
592  {
593  ProtocolLMS64C packet;
594  int inBufPos = 0;
595  pkt.inBuffer.resize(packet.maxDataLength*(length / packet.pktLength + (length % packet.pktLength)), 0);
596  for(int i=0; i<length; i+=packet.pktLength)
597  {
598  pkt.cmd = (eCMD_LMS)buffer[i];
599  pkt.status = (eCMD_STATUS)buffer[i+1];
600  memcpy(&pkt.inBuffer[inBufPos], &buffer[i+8], packet.maxDataLength);
601  inBufPos += packet.maxDataLength;
602  }
603  }
604  else if(protocol == LMS_PROTOCOL_NOVENA)
605  {
606  pkt.cmd = CMD_LMS7002_RD;
607  pkt.status = STATUS_COMPLETED_CMD;
608  pkt.inBuffer.clear();
609  for(int i=0; i<length; i+=2)
610  {
611  //reading from spi returns only registers values
612  //fill addresses as zeros to match generic format of address, value pairs
613  pkt.inBuffer.push_back(0); //should be address msb
614  pkt.inBuffer.push_back(0); //should be address lsb
615  pkt.inBuffer.push_back(buffer[i]);
616  pkt.inBuffer.push_back(buffer[i+1]);
617  }
618  }
619  return 1;

662  sprintf(progressMsg, "Programming failed! Target device not supported");
663  if(callback)
664  abortProgramming = callback(bytesSent, length, progressMsg);
665  return ReportError(ENOTSUP, progressMsg);
666  }
667 
668  unsigned char ctrbuf[64];
669  unsigned char inbuf[64];
670  memset(ctrbuf, 0, 64);
671  ctrbuf[0] = cmd;
672  ctrbuf[1] = 0;
673  ctrbuf[2] = 56;
674 
675  for (portionNumber = 0; portionNumber<portionsCount && !abortProgramming; ++portionNumber)
676  {
677  int offset = 8;
678  memset(&ctrbuf[offset], 0, 56);
679  ctrbuf[offset+0] = prog_mode;
680  ctrbuf[offset+1] = (portionNumber >> 24) & 0xFF;
681  ctrbuf[offset+2] = (portionNumber >> 16) & 0xFF;
682  ctrbuf[offset+3] = (portionNumber >> 8) & 0xFF;
683  ctrbuf[offset+4] = portionNumber & 0xFF;
684  unsigned char data_cnt = data_left > pktSize ? pktSize : data_left;
685  ctrbuf[offset+5] = data_cnt;
686  if(cmd == CMD_MEMORY_WR)
687  {
688  ctrbuf[offset+6] = 0;
689  ctrbuf[offset+7] = 0;
690  ctrbuf[offset+8] = 0;
691  ctrbuf[offset+9] = 0;
692 
693  ctrbuf[offset+10] = (device >> 8) & 0xFF;
694  ctrbuf[offset+11] = device & 0xFF;
695  }
696  if(data_src != NULL)
697  {
698  memcpy(&ctrbuf[offset + 24], data_src, data_cnt);
699  data_src += data_cnt;
700  }
701 
702  if(Write(ctrbuf, sizeof(ctrbuf)) != sizeof(ctrbuf))
703  {
704  if(callback)
705  callback(bytesSent, length, "Programming failed! Write operation failed");
706  return ReportError(EIO, "Programming failed! Write operation failed");
707  }
708  if(Read(inbuf, sizeof(inbuf), progTimeout_ms) != sizeof(ctrbuf))
709  {
710  if(callback)
711  callback(bytesSent, length, "Programming failed! Read operation failed");
712  return ReportError(EIO, "Programming failed! Read operation failed");
713  }
714  data_left -= data_cnt;
715  status = inbuf[1];
716  bytesSent += data_cnt;
717 
718  if(status != STATUS_COMPLETED_CMD)
719  {
720  sprintf(progressMsg, "Programming failed! %s", status2string(status));
721  if(callback)
722  abortProgramming = callback(bytesSent, length, progressMsg);
723  return ReportError(EPROTO, progressMsg);
724  }
725  if(needsData == false) //only one packet is needed to initiate bitstream from flash
726  {
727  bytesSent = length;
728  break;
729  }
730  if(callback)
731  abortProgramming = callback(bytesSent, length, progressMsg);
732  }
733  if (abortProgramming == true)
734  {
735  sprintf(progressMsg, "programming: aborted by user");
736  if(callback)
737  callback(bytesSent, length, progressMsg);
738  return ReportError(ECONNABORTED, "user aborted programming");
739  }
740  sprintf(progressMsg, "programming: completed");
741  if(callback)
742  callback(bytesSent, length, progressMsg);
743 #ifndef NDEBUG
744  auto t2 = std::chrono::high_resolution_clock::now();
745  if ((device == 2 && prog_mode == 2) == false)
746  printf("Programming finished, %li bytes sent! %li ms\n", length, std::chrono::duration_cast<std::chrono::milliseconds>(t2 - t1).count());
747  else
748  printf("FPGA configuring initiated\n");
749 #endif
750  return 0;
751 }

754 {
755  LMS64CProtocol::GenericPacket pkt;
756  pkt.cmd = CMD_ANALOG_VAL_RD;
757 
758  for (size_t i=0; i<count; ++i)
759  pkt.outBuffer.push_back(ids[i]);

761  int status = this->TransferPacket(pkt);

764  assert(pkt.inBuffer.size() >= 4 * count);
765 
766  for (size_t i = 0; i < count; ++i)
767  {
768  int unitsIndex = pkt.inBuffer[i * 4 + 1];
769  if(units)
770  {
771 
772  const char adc_units_prefix[] = {
773  ' ', 'k', 'M', 'G', 'T', 'P', 'E', 'Z',

779 
780  if((unitsIndex & 0xF0)>>4 == TEMPERATURE)
781  values[i] /= 10;
782  }

852 int LMS64CProtocol::ProgramMCU(const uint8_t *buffer, const size_t length, const MCU_PROG_MODE mode, ProgrammingCallback callback)
853 {
854 #ifndef NDEBUG
855  auto timeStart = std::chrono::high_resolution_clock::now();
856 #endif
857  const uint8_t fifoLen = 32;
858  bool success = true;
859  bool terminate = false;
860 
861  int packetNumber = 0;
862  int status = STATUS_UNDEFINED;
863 
864  LMS64CProtocol::GenericPacket pkt;
865  pkt.cmd = CMD_PROG_MCU;
866 
867  if (callback)
868  terminate = callback(0, length,"");
869 
870  for(uint16_t CntEnd=0; CntEnd<length && !terminate; CntEnd+=32)
871  {
872  pkt.outBuffer.clear();
873  pkt.outBuffer.reserve(fifoLen+2);
874  pkt.outBuffer.push_back(mode);
875  pkt.outBuffer.push_back(packetNumber++);
876  for (uint8_t i=0; i<fifoLen; i++)
877  pkt.outBuffer.push_back(buffer[CntEnd + i]);
878 
879  TransferPacket(pkt);
880  status = pkt.status;
881  if (callback)
882  terminate = callback(CntEnd+fifoLen,length,"");
883 #ifndef NDEBUG
884  printf("MCU programming : %4i/%4li\r", CntEnd+fifoLen, long(length));
885 #endif
886  if(status != STATUS_COMPLETED_CMD)
887  {
888  std::stringstream ss;
889  ss << "Programming MCU: status : not completed, block " << packetNumber << std::endl;
890  success = false;
891  break;
892  }
893 
894  if(mode == 3) // if boot mode , send only first packet
895  {
896  if (callback)
897  callback(1, 1, "");
898  break;
899  }
900  };
901 #ifndef NDEBUG
902  auto timeEnd = std::chrono::high_resolution_clock::now();
903  printf("\nMCU Programming finished, %li ms\n",
904  std::chrono::duration_cast<std::chrono::milliseconds>
905  (timeEnd-timeStart).count());
906 #endif
907  return success ? 0 : -1;
908 }
