272 void SoapySDR::Device::setGain(const int dir, const size_t channel, double gain)
273 {
274  //algorithm to distribute overall gain (TX gets BB first, RX gets RF first)
275  const auto names = this->listGains(dir, channel);
276  if (dir == SOAPY_SDR_TX)
277  {
278  for (int i = names.size()-1; i >= 0; i--)
279  {
280  const auto r = this->getGainRange(dir, channel, names[i]);
281  const double g = std::min(gain, r.maximum()-r.minimum());
282  this->setGain(dir, channel, names[i], g+r.minimum());
283  gain -= this->getGain(dir, channel, names[i])-r.minimum();
284  }
285  }
286  if (dir == SOAPY_SDR_RX)
287  {
288  for (size_t i = 0; i < names.size(); i++)
289  {
290  const auto r = this->getGainRange(dir, channel, names[i]);
291  const double g = std::min(gain, r.maximum()-r.minimum());
292  this->setGain(dir, channel, names[i], g+r.minimum());
293  gain -= this->getGain(dir, channel, names[i])-r.minimum();
294  }
295  }
296 }

337 /*******************************************************************
338  * Frequency API
339  ******************************************************************/
340 void SoapySDR::Device::setFrequency(const int dir, const size_t chan, double freq, const Kwargs &args)
341 {
342  const auto comps = this->listFrequencies(dir, chan);
343  if (comps.empty()) return;
344 
345  //optional offset, use on RF element when specified
346  const double offset = (args.count("OFFSET") != 0)?std::atof(args.at("OFFSET").c_str()):0.0;
347 
348  //distribute freq into RF then baseband components
349  for (size_t comp_i = 0; comp_i < comps.size(); comp_i++)
350  {
351  const std::string &name = comps[comp_i];
352 
353  //add offset for RF element (first element)
354  if (comp_i == 0) freq += offset;
355 
356  if (args.count(name) != 0 and args.at(name) == "IGNORE")
357  {
358  //do nothing, dont change component
359  }
360  else if (args.count(name) != 0 and args.at(name) != "DEFAULT")
361  {
362  //specific frequency for component specified
363  const double f(std::atof(args.at(name).c_str()));
364  this->setFrequency(dir, chan, name, f, args);
365  }
366  else
367  {
368  //otherwise use the remainder frequency
369  this->setFrequency(dir, chan, name, freq, args);
370  }
371 
372  //adjust overall freq for the remainder
373  freq -= this->getFrequency(dir, chan, name);
374 
375  //then remove to compensate
376  if (comp_i == 0) freq -= offset;
377  }
378 }

409 SoapySDR::RangeList SoapySDR::Device::getFrequencyRange(const int dir, const size_t chan) const
410 {
411  //get a list of tunable components
412  const auto comps = this->listFrequencies(dir, chan);
413  if (comps.empty()) return SoapySDR::RangeList();
414 
415  //get the range list for the RF component
416  auto ranges = this->getFrequencyRange(dir, chan, comps.front());
417 
418  //use bandwidth setting to clip the range
419  const double bw = this->getBandwidth(dir, chan);
420 
421  //add to the range list given subsequent components
422  for (size_t comp_i = 1; comp_i < comps.size(); comp_i++)
423  {
424  const auto subRange = this->getFrequencyRange(dir, chan, comps[comp_i]);
425  if (subRange.empty()) continue;
426 
427  double subRangeLow = subRange.front().minimum();
428  if (bw > 0.0) subRangeLow = std::max(-bw/2, subRangeLow);
429 
430  double subRangeHigh = subRange.back().maximum();
431  if (bw > 0.0) subRangeHigh = std::min(bw/2, subRangeHigh);
432 
433  for (size_t range_i = 0; range_i < ranges.size(); range_i++)
434  {
435  SoapySDR::Range &range = ranges[range_i];
436  range = SoapySDR::Range(
437  range.minimum() + subRangeLow,
438  range.maximum() + subRangeHigh);
439  }
440  }
441 
442  return ranges;
443 }

450 SoapySDR::ArgInfoList SoapySDR::Device::getFrequencyArgsInfo(const int dir, const size_t chan) const
451 {
452  SoapySDR::ArgInfoList args;
453 
454  const auto comps = this->listFrequencies(dir, chan);
455 
456  if (comps.size() < 2) return args; //no tuning options with single components
457 
458  //support offset tuning
459  {
460  SoapySDR::ArgInfo info;
461  info.key = "OFFSET";
462  info.name = "LO Offset";
463  info.value = "0.0";
464  info.units = "Hz";
465  info.type = SoapySDR::ArgInfo::FLOAT;
466  info.description = "Tune the LO with an offset and compensate with the baseband CORDIC.";
467  SoapySDR::RangeList ranges = this->getFrequencyRange(dir, chan, comps.at(1));
468  if (not ranges.empty()) info.range = ranges.front();
469  args.push_back(info);
470  }
471 
472  //every tunable component becomes an option
473  for (size_t comp_i = 1; comp_i < comps.size(); comp_i++)
474  {
475  SoapySDR::ArgInfo info;
476  info.key = comps[comp_i];
477  info.value = "DEFAULT";
478  info.units = "Hz";
479  info.type = SoapySDR::ArgInfo::FLOAT;
480  info.description = "Specify a specific value for this component or IGNORE to skip tuning it.";
481  info.options.push_back("DEFAULT");
482  info.optionNames.push_back("Default");
483  info.options.push_back("IGNORE");
484  info.optionNames.push_back("Ignore");
485  SoapySDR::RangeList ranges = this->getFrequencyRange(dir, chan, comps.at(comp_i));
486  if (not ranges.empty()) info.range = ranges.front();
487  args.push_back(info);
488  }
489 
490  return args;
491 }
