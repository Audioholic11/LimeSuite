232 void HPM7_wxgui::DownloadAll(wxCommandEvent& event)
233 {
234  if (m_serPort == nullptr || m_serPort->IsOpen() == false)
235  {
236  wxMessageBox(_("Board not connected"), _("Warning"));
237  return;
238  }
239  lime::LMS64CProtocol::GenericPacket pkt;
240  pkt.cmd = lime::CMD_MYRIAD_RD;
241  pkt.outBuffer.push_back(0x10);
242  for (int i = 0; i < 12; ++i)
243  pkt.outBuffer.push_back(0x20 + i);
244  pkt.outBuffer.push_back(0x30);
245  pkt.outBuffer.push_back(0x31);
246 
247  if (m_serPort->TransferPacket(pkt) != 0)
248  {
249  wxMessageBox(_("Failed to read board parameters"));
250  return;
251  }
252 
253  assert(pkt.inBuffer.size() >= 14);
254  cmbActivePath->SetSelection(pkt.inBuffer[1] & 0x3);
255  cmbBand->SetSelection((pkt.inBuffer[1] >> 2) & 0x1);
256  cmbLNA->SetSelection((pkt.inBuffer[1] >> 3) & 0x1);
257  cmbPAdriver->SetSelection((pkt.inBuffer[1] >> 4) & 0x1);
258 
259  int index = 3;
260  for (unsigned i = 0; i < chkEB.size(); ++i)
261  {
262  cmbSSC1[i]->SetSelection(pkt.inBuffer[index] & 0x1F);
263  index+=2;
264  chkEB[i]->SetValue((pkt.inBuffer[index] >> 5) & 1);
265  chkTP[i]->SetValue((pkt.inBuffer[index] >> 4) & 1);
266  cmbSSC2[i]->SetSelection(pkt.inBuffer[index] & 0xF);
267  index += 2;
268  }
269 
270  cmbDAC_A->SetSelection(pkt.inBuffer[index]);
271  index += 2;
272 cmbDAC_B->SetSelection(pkt.inBuffer[index]);
273 }

312 bool HPM7_wxgui::UploadGPIO()
313 {
314  if (m_serPort == nullptr || m_serPort->IsOpen() == false)
315  {
316  wxMessageBox(_("Uploading HPM7 GPIO, board not connected"), _("Warning"));
317  return false;
318  }
319 
320  lime::LMS64CProtocol::GenericPacket pkt;
321  pkt.cmd = lime::CMD_MYRIAD_WR;
322  pkt.outBuffer.push_back(0x10);
323  unsigned char value = 0;
324  int activePath = cmbActivePath->GetSelection();
325  switch (activePath)
326  {
327  case 0: value |= 0; break;
328  case 1: value |= 2; break;
329  case 2: value |= 3; break;
330  case 3: value |= 1; break;
331  }
332  value |= (cmbBand->GetSelection() & 0x1) << 2;
333  value |= (cmbLNA->GetSelection() & 0x1) << 3;
334  value |= (cmbPAdriver->GetSelection() & 0x1) << 4;
335  pkt.outBuffer.push_back(value);
336  if (m_serPort->TransferPacket(pkt) != 0)
337  {
338  wxMessageBox(_("Uploading HPM7 GPIO failed"), _("Warning"));
339  return false;
340  }
341 return true;
